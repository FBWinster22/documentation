import { Injectable } from '@angular/core';
import { AgSnackbarService } from "../services/ag-snackbar.service";

@Injectable({
  providedIn: 'root'
})
export class FormValidationService {
  
  constructor( private snackBar: AgSnackbarService ) { }

  // Validación principal de formularios
  validateMainForm(apiResponse: any, form_object: any): boolean {
    let isValid = true;
    let contador_campos = 0

    apiResponse?.content.forEach((element: any, index: number) => {
        const value = form_object[element.name];

        // Validación de campo obligatorio
        if (element.required === 1 && value === '') {
            this.snackBar.error(`El campo ${element.label} es obligatorio`);
            apiResponse!.content[contador_campos]['isValid'] = false;
            isValid = false;
        }

        // Validaciones adicionales
        if (isValid && element.validation && element.validation !== '') {
            const parsed = JSON.parse(element.validation);

            if (parsed) {
                const validators = parsed;

                // Validación de longitud mínima
                if (validators.min_length !== undefined) {
                    if (!this.validateMinLength(value, validators.min_length)) {
                        this.snackBar.error(`El campo ${element.label} debe tener una longitud mínima de ${validators.min_length}`);
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }

                // Validación de expresión regular con mensaje personalizado
                if (validators.reg_exp !== undefined) {
                    const errorMsg = validators.reg_exp_msg || `El campo ${element.label} no cumple con el patrón requerido`;
                    if (!this.validateRegExp(value, validators.reg_exp, errorMsg)) {
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }

                // Validación de valor mayor que
                if (validators.greater_than !== undefined) {
                    if (!this.validateGreaterThan(+value, validators.greater_than)) {
                        this.snackBar.error(`El campo ${element.label} debe ser mayor que ${validators.greater_than}`);
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }

                // Validación de valor menor que
                if (validators.lower_than !== undefined) {
                    if (!this.validateLowerThan(+value, validators.lower_than)) {
                        this.snackBar.error(`El campo ${element.label} debe ser menor que ${validators.lower_than}`);
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }

                // Validación de fecha mayor que hoy
                if (validators.date_gt_today !== undefined) {
                    if (!this.validateDateGreaterThanToday(value)) {
                        this.snackBar.error(`El campo ${element.label} debe ser una fecha posterior a hoy`);
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }

                // Validación de fecha menor que hoy
                if (validators.date_lt_today !== undefined) {
                    if (!this.validateDateLowerThanToday(value)) {
                        this.snackBar.error(`El campo ${element.label} debe ser una fecha anterior a hoy`);
                        apiResponse!.content[contador_campos]['isValid'] = false;
                        isValid = false;
                    }
                }
            }
        }
        contador_campos++
    });
    
    return isValid;
  }

  // Validación de formularios relacionales
  validateRelationalForms(relationalData: any, form_object_relational: any): boolean {
    let isValid = true;

    if(relationalData != undefined){
      relationalData.forEach((relation: any, i: number) => {
        relation.relationals.forEach((row: any, j: number) => {
            row.forEach((item: any, k: number) => {
                const value = form_object_relational[i][j][item.name];

                // Validación de campo obligatorio
                if (item.required === 1 && value === '') {
                    this.snackBar.error(`El campo ${item.label} es obligatorio`);
                    relationalData[i].relationals[j][k]['isValid'] = false
                    isValid = false;
                }

                // Validaciones adicionales
                if (isValid && item.validation && item.validation !== '') {
                    const parsed = JSON.parse(item.validation);

                    if (parsed) {
                        const validators = parsed;

                        // Validación de longitud mínima
                        if (validators.min_length !== undefined) {
                            if (!this.validateMinLength(value, validators.min_length)) {
                                this.snackBar.error(`El campo ${item.label} debe tener una longitud mínima de ${validators.min_length}`);
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }

                        // Validación de expresión regular con mensaje personalizado
                        if (validators.reg_exp !== undefined) {
                            const errorMsg = validators.reg_exp_msg || `El campo ${item.label} no cumple con el patrón requerido`;
                            if (!this.validateRegExp(value, validators.reg_exp, errorMsg)) {
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }

                        // Validación de valor mayor que
                        if (validators.greater_than !== undefined) {
                            if (!this.validateGreaterThan(+value, validators.greater_than)) {
                                this.snackBar.error(`El campo ${item.label} debe ser mayor que ${validators.greater_than}`);
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }

                        // Validación de valor menor que
                        if (validators.lower_than !== undefined) {
                            if (!this.validateLowerThan(+value, validators.lower_than)) {
                                this.snackBar.error(`El campo ${item.label} debe ser menor que ${validators.lower_than}`);
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }

                        // Validación de fecha mayor que hoy
                        if (validators.date_gt_today !== undefined) {
                            if (!this.validateDateGreaterThanToday(value)) {
                                this.snackBar.error(`El campo ${item.label} debe ser una fecha posterior a hoy`);
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }

                        // Validación de fecha menor que hoy
                        if (validators.date_lt_today !== undefined) {
                            if (!this.validateDateLowerThanToday(value)) {
                                this.snackBar.error(`El campo ${item.label} debe ser una fecha anterior a hoy`);
                                relationalData[i].relationals[j][k]['isValid'] = false
                                isValid = false;
                            }
                        }
                    }
                }
            });
        });
    });
    }
    

    return isValid;
  }

  // Validación de longitud mínima
  private validateMinLength(value: string, minLength: number): boolean {
    return value.length >= minLength;
  }

  // Validación de expresión regular con mensaje personalizado
  private validateRegExp(value: string, pattern: string, errorMsg: string): boolean {
    const regExp = new RegExp(pattern);
    const isValid = regExp.test(value);
    
    if (!isValid) {
      this.snackBar.error(errorMsg);
    }
    
    return isValid;
  }

  // Validación de valor mayor que
  private validateGreaterThan(value: number, threshold: number): boolean {
    return value > threshold;
  }

  // Validación de valor menor que
  private validateLowerThan(value: number, threshold: number): boolean {
    return value < threshold;
  }

  // Validación de fecha mayor que hoy
  private validateDateGreaterThanToday(value: string): boolean {
    const dateValue = new Date(value);
    const today = new Date();
    return dateValue > today;
  }

  // Validación de fecha menor que hoy
  private validateDateLowerThanToday(value: string): boolean {
    const dateValue = new Date(value);
    const today = new Date();
    return dateValue < today;
  }

  // Validación de unicidad (requiere comparación con una lista de valores existentes)
  private validateUnique(value: string, existingValues: string[]): boolean {
    return !existingValues.includes(value);
  }

  // Validación de campos iguales
  private validateFieldsEqual(value1: string, value2: string): boolean {
    return value1 === value2;
  }

  // Validación de número entero
  private validateInteger(value: string): boolean {
    return /^\d+$/.test(value);
  }

  // Validación de número decimal
  private validateDecimal(value: string): boolean {
    return /^\d+(\.\d+)?$/.test(value);
  }

  // Validación de rango de fechas
  private validateDateRange(value: string, startDate: string, endDate: string): boolean {
    const dateValue = new Date(value);
    const start = new Date(startDate);
    const end = new Date(endDate);
    return dateValue >= start && dateValue <= end;
  }

  // Validación de no nulo o no vacío
  private validateNotNullOrEmpty(value: any): boolean {
    return value !== null && value !== undefined && value !== '';
  }

  // Validación de solo letras
  private validateOnlyLetters(value: string): boolean {
    return /^[A-Za-z]+$/.test(value);
  }

  // Validación de solo letras y espacios
  private validateOnlyLettersAndSpaces(value: string): boolean {
    return /^[A-Za-z\s]+$/.test(value);
  }

  // Validación de número de teléfono
  private validatePhoneNumber(value: string): boolean {
    return /^\+?[1-9]\d{1,14}$/.test(value);
  }

  // Validación de URL
  private validateUrl(value: string): boolean {
    const urlPattern = new RegExp('^(https?:\\/\\/)?'+ // validate the protocol
                                  '((([a-z\\d]([a-z\\d-]*[a-z\\d])*)\\.?)+[a-z]{2,}|'+ // validate domain name
                                  '((\\d{1,3}\\.){3}\\d{1,3}))'+ // OR validate IP address (IPv4)
                                  '(\\:\\d+)?(\\/[-a-z\\d%_.~+]*)*'+ // validate port and path
                                  '(\\?[;&a-z\\d%_.~+=-]*)?'+ // validate query string
                                  '(\\#[-a-z\\d_]*)?$','i'); // validate fragment locator
    return urlPattern.test(value);
  }
}
