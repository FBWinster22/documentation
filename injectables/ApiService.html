import { HttpClient, HttpHeaders, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';

/**
 * Servicio para realizar solicitudes HTTP con manejo de tokens de autenticación.
 * 
 * Este servicio proporciona métodos para realizar solicitudes HTTP GET, POST, PUT y DELETE, y maneja el token de autenticación
 * necesario para las solicitudes. El token se puede establecer manualmente o extraer de localStorage.
 */
@Injectable({
  providedIn: 'root'
})
export class ApiService {
  private authToken: string = 'e9ada58e-ad7d-4a09-a130-a8ade4108c49723642f9-27a1-4be5-bfac-c2925d2a8f40ed12f1ae-e38a-4c4d-9595-e274248f3960';

  /**
   * Crea una instancia del servicio ApiService.
   * 
   * @param {HttpClient} http - El cliente HTTP para realizar solicitudes.
   */
  constructor(private http: HttpClient) { }

  /**
   * Establece el token de autenticación para las solicitudes HTTP.
   * 
   * @param {string} token - El token de autenticación a establecer.
   */
  setAuthToken(token: string) {
    this.authToken = token;
  }

  /**
   * Realiza una solicitud HTTP GET.
   * 
   * @param {string} url - La URL para la solicitud GET.
   * @param {HttpParams} [params] - Parámetros opcionales para la solicitud.
   * @param {HttpHeaders} [headers] - Encabezados opcionales para la solicitud.
   * @returns {Observable<T>} Un observable con la respuesta de la solicitud.
   * @template T - El tipo de respuesta esperado.
   */
  get<T>(url: string, params?: HttpParams, headers?: HttpHeaders): Observable<T> {
    const allHeaders = this.createHeaders(headers);
    // return this.http.get<T>(url, { headers: allHeaders, params: params});
    return this.http.get<any>(url, {
      headers: allHeaders
    });

  }

   /**
   * Realiza una solicitud HTTP POST.
   * 
   * @param {string} url - La URL para la solicitud POST.
   * @param {any} body - El cuerpo de la solicitud POST.
   * @param {Object} [options] - Opciones adicionales para la solicitud, incluyendo encabezados y parámetros.
   * @param {HttpHeaders} [options.headers] - Encabezados opcionales para la solicitud.
   * @param {HttpParams} [options.params] - Parámetros opcionales para la solicitud.
   * @returns {Observable<T>} Un observable con la respuesta de la solicitud.
   * @template T - El tipo de respuesta esperado.
   */
  post<T>(url: string, body: any, options?: { headers?: HttpHeaders, params?: HttpParams }): Observable<T> {
    const allHeaders = this.createHeaders(options?.headers);
    return this.http.post<any>(url, body, { headers: {
      'Authorization' : `${this.authToken}` 
    }, ...options });
  }
  
  /**
   * Realiza una solicitud HTTP PUT.
   * 
   * @param {string} url - La URL para la solicitud PUT.
   * @param {any} body - El cuerpo de la solicitud PUT.
   * @param {Object} [options] - Opciones adicionales para la solicitud, incluyendo encabezados y parámetros.
   * @param {HttpHeaders} [options.headers] - Encabezados opcionales para la solicitud.
   * @param {HttpParams} [options.params] - Parámetros opcionales para la solicitud.
   * @returns {Observable<T>} Un observable con la respuesta de la solicitud.
   * @template T - El tipo de respuesta esperado.
   */
  put<T>(url: string, body: any, options?: { headers?: HttpHeaders, params?: HttpParams }): Observable<T> {
    const allHeaders = this.createHeaders(options?.headers);
    return this.http.put<any>(url, body, { headers: {
      'Authorization' : `${this.authToken}` 
    }, ...options });
  }

  /**
   * Realiza una solicitud HTTP DELETE.
   * 
   * @param {string} url - La URL para la solicitud DELETE.
   * @param {Object} [options] - Opciones adicionales para la solicitud, incluyendo encabezados y parámetros.
   * @param {HttpHeaders} [options.headers] - Encabezados opcionales para la solicitud.
   * @param {HttpParams} [options.params] - Parámetros opcionales para la solicitud.
   * @returns {Observable<T>} Un observable con la respuesta de la solicitud.
   * @template T - El tipo de respuesta esperado.
   */
  delete<T>(url: string, options?: { headers?: HttpHeaders, params?: HttpParams }): Observable<T> {
    const allHeaders = this.createHeaders(options?.headers);

    return this.http.delete<T>(url, {
        headers: allHeaders.set('Authorization', `${this.authToken}`),
        params: options?.params,
    });
}
  /**
   * Crea los encabezados HTTP necesarios para las solicitudes.
   * 
   * Este método incluye el token de autenticación en los encabezados si está disponible.
   * 
   * @param {HttpHeaders} [headers] - Encabezados opcionales a combinar con los encabezados creados.
   * @returns {HttpHeaders} Los encabezados HTTP combinados.
   */
  private createHeaders(headers?: HttpHeaders): HttpHeaders {
    let allHeaders = new HttpHeaders();
    
    if (this.authToken) {
      allHeaders = allHeaders.append('Authorization', `${this.authToken}`);
    }

    if (headers) {
      headers.keys().forEach(key => {
        allHeaders = allHeaders.append(key, headers.get(key) as string);
      });
    }

    return allHeaders;
  }

  // Método para establecer el token de autorización desde cualquier lugar de la aplicación

  /**
   * Establece el token de autorización desde localStorage.
   * 
   * Este método recupera el token de autorización desde localStorage y lo establece en el servicio.
   */
  setAuthorizationTokenFromLocalStorage() {
    const token = localStorage.getItem('token');
    if (token) {
      this.setAuthToken(token);
    }
  }
}
