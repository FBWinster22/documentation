import { Component, ViewChild, viewChild } from '@angular/core';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { AgPopupAlertComponent } from '../../../widgets/ag-popup-alert/ag-popup-alert.component';
import { environment } from '../../../../../environments/environment.development';
import { AgSearchComponent } from "../../../widgets/ag-search/ag-search.component";
import { AgSelectComponent } from "../../../widgets/ag-select/ag-select.component";
import { MatIcon } from '@angular/material/icon';
import { ApiService } from '../../../functions/api.service';

/**
 * Componente para la creación de formularios.
 * 
 * Maneja la descarga del esquema de campos de un formulario y su adición a un componente de formulario.
 * 
 * @component
 */
@Component({
  selector: 'ag-create-form',
  standalone: true,
  imports: [AgFormComponent, AgPopupAlertComponent, AgSearchComponent, AgSelectComponent, MatIcon],
  templateUrl: './ag-create-form.component.html',
  styleUrl: './ag-create-form.component.css'
})
export class AgCreateFormComponent {

  /**
   * Referencia al componente de formulario utilizado para gestionar los campos y su esquema.
   * 
   * @type {AgFormComponent | undefined}
   */
  @ViewChild('FORM_TABLE_FORMS') FORM_TABLE_FORMS?: AgFormComponent

  /**
   * Variable que contiene la configuración del entorno.
   * 
   * @type {any}
   */
  environment = environment

  /**
   * Nombre de la tabla seleccionada para la descarga del esquema de campos.
   * 
   * @type {string | undefined}
   */
  selected_table?: string

  /**
   * Constructor del componente.
   * 
   * @param {ApiService} apiService - Servicio para la comunicación con la API.
   */
  constructor(private apiService: ApiService){
  }

  /**
   * Descarga el esquema de campos para la tabla seleccionada y actualiza el componente de formulario.
   * 
   * Realiza una solicitud a la API para obtener el esquema de campos y los valores asociados.
   * Luego, agrega estos campos al componente de formulario.
   * 
   * @memberof AgCreateFormComponent
   */
  descargarEsquema(){ //Nombre inconsistente

    this.apiService.get(environment.URL_BACKEND+'/form_fields/generate-by-table?sql_table_name='+this.selected_table).subscribe((data: any)=>{

      // a este this.FORM_TABLE_FORMS?.form_object_relational[0] hay que agregarle cositas

      //agregar primero el esquema de form fields, y luego los valores...

      let i = 0

      while( i < data.form_fields_list.length){

        data.form_fields_list[i]['options'] = 'blank'
        data.form_fields_list[i]['automated'] = 1
        data.form_fields_list[i]['hidden'] = 0
        
        this.FORM_TABLE_FORMS?.addRelationalItem(data.form_field_schema,0,data.form_fields_list[i])
        i++
      }


    })

  }

}
