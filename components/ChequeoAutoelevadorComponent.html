Description

Componente ChequeoAutoelevadorComponent encargado de gestionar y visualizar los chequeos mensuales y diarios de autoelevadores.

Este componente permite cargar datos de chequeos desde una API, actualizarlos en formularios, y mostrar los resultados en tablas. Además, maneja la lógica para verificar el vencimiento de los chequeos y proporciona retroalimentación al usuario mediante un snackBar.

import { Component, OnInit, ViewChild } from '@angular/core';
import { AgFormComponent } from '../lib/widgets/ag-form/ag-form.component';
import { AgInputComponent } from '../lib/widgets/ag-input/ag-input.component';
import { ApiService } from '../lib/functions/api.service';
import { CommonModule } from '@angular/common';
import { MatSnackBar, MatSnackBarModule } from '@angular/material/snack-bar';
import { AgTableComponent } from '../lib/widgets/ag-table/ag-table.component';
import { environment } from '../../environments/environment.development';
import { ApiSearchInterface } from '../lib/interfaces/ApiSearchGroupInterface';

/**
 * Componente `ChequeoAutoelevadorComponent` encargado de gestionar y visualizar
 * los chequeos mensuales y diarios de autoelevadores.
 * 
 * Este componente permite cargar datos de chequeos desde una API, actualizarlos
 * en formularios, y mostrar los resultados en tablas. Además, maneja la lógica
 * para verificar el vencimiento de los chequeos y proporciona retroalimentación
 * al usuario mediante un `snackBar`.
 * 
 * @component
 * @implements OnInit
 */
@Component({
  selector: 'app-chequeo-autoelevador',
  standalone: true,
  imports: [AgFormComponent, AgInputComponent, CommonModule, MatSnackBarModule, AgTableComponent],
  templateUrl: './chequeo-autoelevador.component.html',
  styleUrl: './chequeo-autoelevador.component.css'
})
export class ChequeoAutoelevadorComponent implements OnInit{

  /**
 * Crea una instancia del componente `ChequeoAutoelevadorComponent`.
 * 
 * @param {ApiService} apiService - Servicio para realizar peticiones HTTP a la API.
 * @param {MatSnackBar} snackBar - Servicio para mostrar mensajes emergentes (`snack-bar`).
 */
  constructor(private apiService:ApiService, private snackBar: MatSnackBar){}

  /**
   * URL del backend donde se realizan las peticiones API.
   * @type {string}
   */
  urlBackend:String = "http://127.0.0.1:8000/"

  /**
   * Almacena los resultados obtenidos de la API.
   * @type {any[]}
   */
  results: any[] = [];

  /**
   * Resultados para la tabla de chequeos mensuales.
   * @type {any[]}
   */
  resultsTableMensual:any[] = [];
  /**
   * Resultados para la tabla de chequeos diarios.
   * @type {any[]}
   */
  resultsTableDiaria: any[] = [];

  /**
   * Almacena la configuración de búsqueda para la tabla mensual de chequeos.
   * @type {ApiSearchInterface | undefined}
   */
  tabla_mensual_chequeo_api_search_group?: ApiSearchInterface;

  /**
   * Almacena la configuración de búsqueda para la tabla de chequeos.
   * @type {ApiSearchInterface | undefined}
   */
  tabla_chequeo_api_search_group?: ApiSearchInterface;

  /**
   * Controla la visibilidad de la tabla mensual de chequeos.
   * @type {boolean}
   */
  mostrarResultadosTablaMensualChequeo: boolean = true; //Nombre inconsistente

  /**
   * Acceso a las variables de entorno.
   * @type {typeof environment}
   */
  environment = environment;

  /**
   * Indica si los chequeos están vencidos.
   * @type {boolean | undefined}
   */
  isVencido?:boolean; //Nombre inconsistente

  /**
   * Indica si el formulario fue enviado correctamente.
   * @type {boolean | undefined}
   */
  isSend?:boolean;

  /**
   * Controla la visualización de registros.
   * @type {boolean}
   */
  isRegistros?:boolean = true; //Nombre inconsistente

  /**
   * Referencia al formulario de chequeos de autoelevadores.
   * @type {AgFormComponent | undefined}
   */
  @ViewChild('FORM_TABLE_CHEQUEO_AUTOELEVADOR') FORM_TABLE_CHEQUEO_AUTOELEVADOR?: AgFormComponent

  /**
   * Referencia al formulario mensual de chequeos de autoelevadores.
   * @type {AgFormComponent | undefined}
   */
  @ViewChild('FORM_MENSUAL_TABLE_CHEQUEO_AUTOELEVADOR') FORM_MENSUAL_TABLE_CHEQUEO_AUTOELEVADOR?: AgFormComponent

  /**
   * Referencia a la tabla mensual de chequeos de autoelevadores.
   * @type {AgTableComponent | undefined}
   */
  @ViewChild('tabla_mensual_chequeo_autoelevador') tabla_mensual_chequeo_autoelevador?: AgTableComponent

  /**
   * Referencia a la tabla de chequeos de autoelevadores.
   * @type {AgTableComponent | undefined}
   */
  @ViewChild('tabla_chequeo_autoelevador') tabla_chequeo_autoelevador?: AgTableComponent

  /**
   * Objeto para almacenar los parámetros de búsqueda en la API.
   * @type {Object}
   */
  body:any = {
    page: 1,
    criteria_group: {
      group_list: [
        {
          logic_operator: "AND", // Ajusta esto según tus necesidades
          criteria_list: [
            {
              column_filter: '',
              compare_criteria: "PARECIDO A", // Ajusta esto según tus necesidades
              compare_value: ''
            }
          ],
          group_list: []
        }
      ]
    }
  };

  /**
   * Carga los datos en el formulario de chequeo mensual y muestra el formulario.
   * 
   * @param {any} event - Datos del evento para cargar en el formulario.
   * @memberof ChequeoAutoelevadorComponent
   */
  loadIntoForm(event: any){
    this.reveal_form = true
    setTimeout(() => {
      this.FORM_MENSUAL_TABLE_CHEQUEO_AUTOELEVADOR!.form_object = event
    }, 250);
  }

  /**
   * Actualiza la configuración de búsqueda para la tabla mensual de chequeos
   * y realiza una nueva búsqueda.
   * 
   * @param {ApiSearchInterface} apiSearchInterface - Configuración de búsqueda para la tabla mensual.
   * @memberof ChequeoAutoelevadorComponent
   */
  update_tabla_mensual_chequeo_search_interface(apiSearchInterface: any){

    this.tabla_mensual_chequeo_api_search_group = apiSearchInterface
    this.tabla_mensual_chequeo_autoelevador!.currentPage = 1
    this.tabla_mensual_chequeo_autoelevador?.callSearch()

  }

  /**
   * Actualiza la configuración de búsqueda para la tabla de chequeos
   * y realiza una nueva búsqueda.
   *
   * @memberof ChequeoAutoelevadorComponent 
   * @param {ApiSearchInterface} apiSearchInterface - Configuración de búsqueda para la tabla de chequeos.
   */
  update_tabla_hequeo_search_interface(apiSearchInterface: any){ //Nombre inconsistente

    this.tabla_chequeo_api_search_group = apiSearchInterface
    this.tabla_chequeo_autoelevador!.currentPage = 1
    this.tabla_chequeo_autoelevador?.callSearch()

  }

  reveal_form = false;

  /**
   * Muestra un mensaje emergente indicando si el chequeo fue cargado correctamente.
   * 
   * @param {boolean} isSend - Indica si el formulario fue enviado correctamente.
   * @memberof ChequeoAutoelevadorComponent
   */
  openSnackBar(isSend:boolean) {
    if(isSend)
      this.snackBar.open('CHEQUEO CARGADO CORRECTAMENTE', 'Cerrar', {
        duration: 3000, // Duración en milisegundos (opcional)
    });
  }

  /**
   * Maneja el envío del formulario y muestra un mensaje emergente.
   * 
   * @param {boolean} value - Indica si el formulario fue enviado.
   * @memberof ChequeoAutoelevadorComponent
   */
  submitForm(value:boolean){
    this.isSend = value;
    this.openSnackBar(this.isSend)
  }

  /**
   * Método llamado al inicializar el componente.
   * Realiza una petición a la API para obtener los chequeos mensuales
   * y verifica si están vencidos.
   * 
   * @memberof ChequeoAutoelevadorComponent
   */
  ngOnInit(): void {
    const apiUrl = `${this.urlBackend}api/search/mensual_chequeo_autoelevador`;

    this.apiService.post(apiUrl, this.body).subscribe(
      (data: any) => {
        this.results = data.results; // Ajusta según la estructura de respuesta de tu API
        this.verVencimientoMensual(this.results)
      },
      error => {
        console.error('Error fetching data', error);
      }
    );
  }

  /**
   * Verifica si alguno de los chequeos mensuales está vencido.
   * 
   * @param {any[]} results - Lista de resultados de chequeos mensuales.
   * @memberof ChequeoAutoelevadorComponent
   */
  verVencimientoMensual(results: any[]) { //Nombre inconsistente
    if (!results || results.length === 0) {
      this.isVencido = true; // Si no hay resultados, no está vencido
      return;
    }
  
    const haceUnMes = new Date();//NOmbre inconsistente
    haceUnMes.setMonth(haceUnMes.getMonth() - 1); // Añadir un mes a la fecha actual
  
    this.isVencido = false; // Inicializar como no vencido
  
    for (let item of results) {
      const fechaConvertida = new Date(item.fechahora); // Convertir item.fechahora a un objeto Date
  
      if (fechaConvertida <= haceUnMes) {
        this.isVencido = true; // Si al menos uno está vencido, establecer isVencido y salir
      }
      else
        this.isVencido = false;
    }
  
    // Si ninguno está vencido
    
  }

}
