Description

Componente de búsqueda con opciones avanzadas y una tabla para mostrar los resultados.

El componente AgSearchComponent proporciona una interfaz para realizar búsquedas avanzadas con diferentes operadores de búsqueda. Permite al usuario seleccionar ítems y mostrar resultados en una tabla. Incluye iconos para varias acciones y puede ser configurado para mostrar una forma simplificada o completa.

    /**
 * Inicializa el componente cargando los datos de la tabla y configurando el formulario de búsqueda.
 * 
 * Este método se llama automáticamente cuando el componente se inicializa. Realiza una solicitud a la API para obtener la configuración de los encabezados de la tabla usando el `tableKeyName` proporcionado. Luego, procesa la respuesta para configurar las cabeceras de la tabla y llama al método `addGroupList` para completar la configuración del componente.

  }

  /**

  buscar() { // Nombre de método inconsistente

    setTimeout(() => {
      // this.onAPIResponse.emit(this.api_search_object ? this.api_search_object : getBaseApiSearchInterface())
      this.api_search_criteria_group = this.api_search_object ? this.api_search_object : getBaseApiSearchInterface()
      this.tableId.api_search_criteria_group = this.api_search_criteria_group
      this.tableId.callSearch()
    }, 100);

    this.criteriaDescriptions = this.mapCriteriaGroup(this.api_search_object.criteria_group)
  }

  /**
 * Maneja la selección de un ítem y actualiza el formulario con los detalles del ítem seleccionado.

  onSelectedItem(event: any) {
    // console.log(event)

    // let y = 500;

    // // window.mo
    // this.contextMenuService.showMenu(x,y)

    // return

    this.onItemSelected?.emit(event)

    if (this.formRef == undefined) return
    this.formRef!.cleanForm()
    this.formRef!.reveal = true
    const eventCopy = { ...event }; // Hacer una copia del objeto event

    setTimeout(() => {
      // this.formRef!.form_object = event
      this.formRef!.form_object = eventCopy

      setTimeout(() => {
        this.formRef!.getRelatedItems()
      }, 100);
    }, 250);
  }

  /**
 * Mapea y describe los criterios de búsqueda en un formato legible.
 * 
 * Este método convierte el grupo de criterios de búsqueda en una lista de descripciones en formato de texto. Utiliza la función `mapCriteria` para procesar los criterios y subgrupos de criterios, concatenando las descripciones y aplicando el operador lógico adecuado entre ellos.
 * 
 * @param {BaseApiSearchGroupInterface} criteriaGroup - El grupo de criterios que se va a mapear.
 * @returns {string[]} - Un array de descripciones en formato de texto de los criterios de búsqueda.
 * @memberof AgSearchComponent
 * @method mapCriteriaGroup
 */
  mapCriteriaGroup(criteriaGroup: BaseApiSearchGroupInterface): string[] {
    const criteriaDescriptions: string[] = [];

    const mapCriteria = (group: BaseApiSearchGroupInterface, parentLogicOperator?: string) => {
      const localDescriptions: string[] = [];

      group.criteria_list.forEach((criteria) => {
        localDescriptions.push(
          `${criteria.column_filter} ${criteria.compare_criteria} ${criteria.compare_value}`
        );
      });

      group.group_list.forEach((subGroup) => {
        localDescriptions.push(`${mapCriteria(subGroup, group.logic_operator).join(' ' + '/-' + ' ')}`);
      });

      if (parentLogicOperator && parentLogicOperator !== group.logic_operator) {
        return [`${localDescriptions.join(' ' + '/-' + ' ')}`];
      }

      return localDescriptions;
    };

    criteriaDescriptions.push(...mapCriteria(criteriaGroup));
    return criteriaDescriptions;
  }
}
