import { CommonModule } from '@angular/common';
import { Component } from '@angular/core';
import { AgLoginFormComponent } from './components/ag-form-login/ag-form-login.component';
import { AgLogoComponent } from '../ag-logo/ag-logo.component';
import { AgRegisterFormComponent } from './components/ag-form-register/ag-form-register.component';

/**
 * Componente para el inicio de sesión y registro de usuarios.
 * 
 * Este componente proporciona una interfaz para que los usuarios inicien sesión o se registren en la aplicación.
 * 
 * @component
 * @memberof AgLoginComponent
 */
@Component({
  selector: 'ag-login',
  standalone: true,
  imports: [CommonModule, AgLoginFormComponent, AgRegisterFormComponent ,AgLogoComponent],
  templateUrl: './ag-login.component.html',
  styleUrl: './ag-login.component.css'
})
export class AgLoginComponent {

  /**
   * Indica si el formulario de registro está activo.
   * 
   * @type {boolean}
   */
  isNew:boolean = false

  /**
   * Alterna entre el formulario de inicio de sesión y el de registro.
   * 
   * Cambia el estado de `isNew` entre `true` y `false`, permitiendo la visualización del formulario de registro o de inicio de sesión.
   * 
   * @method
   * @memberof AgLoginComponent
   */
  changeForm(){
    this.isNew = !this.isNew
  }
}
<div class="login_container">
    <div class="login_content">
        <div *ngIf="isNew" class="form_reg">
            <ag-register-form (isNew)="isNew = $event"></ag-register-form>
        </div>
        <div [ngClass]="isNew ? 'logo_left anim_left' : 'logo_right anim_right'">
            <app-ag-logo></app-ag-logo>
        </div>
        <div *ngIf="!isNew" class="form_log">
            <ag-login-form (isNew)="isNew = $event"></ag-login-form>
        </div>
    </div>
</div>
