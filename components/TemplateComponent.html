import { Component, ViewChild } from '@angular/core';
import { AgSearchComponent } from '../ag-search/ag-search.component';
import { AgFormComponent } from '../ag-form/ag-form.component';
import { CommonModule } from '@angular/common';

/**
 * Componente `TemplateComponent` se utiliza para manejar plantillas de formularios y búsqueda.
 * 
 * Este componente se encarga de integrar los componentes `AgSearchComponent` y `AgFormComponent`,
 * y proporciona funcionalidad para interactuar con formularios y búsquedas dentro de la aplicación.
 * 
 * @component
 */
@Component({
  selector: 'app-template',
  standalone: true,
  imports: [AgSearchComponent, AgFormComponent, CommonModule],
  templateUrl: './template.component.html',
  styleUrl: './template.component.css'
})
export class TemplateComponent {

  /**
   * Referencia al componente de formulario `AgFormComponent` para la tabla de paquetes ISG.
   * 
   * @type {AgFormComponent}
   * @memberof TemplateComponent
   * @viewChild FORM_TABLE_PAQUETESISG
   */
  @ViewChild('FORM_TABLE_PAQUETESISG') formTablePaqueteISG !: AgFormComponent
}
