import { Component, EventEmitter, OnInit, Output, ViewChild } from '@angular/core';
import { ApiService } from '../../../../../functions/api.service';
import { HttpHeaders } from '@angular/common/http';
import { FormsModule } from '@angular/forms';
import { environment } from '../../../../../../../environments/environment.development';
import { SignInfoData } from '../../../../../interfaces/APIResponseInterface';
import { AgInputComponent } from '../../../../../widgets/ag-input/ag-input.component';
import { AgSocialMediaComponent } from '../../../../../widgets/ag-social-media/ag-social-media.component';
import { Router } from '@angular/router';

/**
 * Componente para el formulario de inicio de sesión.
 * 
 * Este componente proporciona una interfaz para que los usuarios ingresen su nombre de usuario y contraseña,
 * manejando la autenticación mediante la comunicación con una API.
 * 
 * @component
 * @memberof AgLoginFormComponent
 */
@Component({
  selector: 'ag-login-form',
  standalone: true,
  imports: [AgInputComponent ,AgSocialMediaComponent, FormsModule],
  templateUrl: './ag-form-login.component.html',
  styleUrl: './ag-form-login.component.css'
})
export class AgLoginFormComponent implements OnInit{

  /**
   * Referencia al componente de entrada para el campo de correo electrónico.
   * 
   * @type {AgInputComponent | undefined}
   * @memberof AgLoginFormComponent
   */
  @ViewChild('email') email ?: AgInputComponent

  /**
   * Referencia al componente de entrada para el campo de contraseña.
   * 
   * @type {AgInputComponent | undefined}
   * @memberof AgLoginFormComponent
   */
  @ViewChild('pass') pass ?: AgInputComponent

  /**
   * Emisor de eventos que indica si se debe cambiar al formulario de registro.
   * 
   * @type {EventEmitter<boolean>}
   * @memberof AgLoginFormComponent
   */
  @Output() isNew: EventEmitter<boolean> = new EventEmitter<boolean>()

  /**
   * Datos recibidos del servidor relacionados con la autenticación.
   * 
   * @type {SignInfoData}
   * @memberof AgLoginFormComponent
   */
  resData!: SignInfoData

  /**
   * Objeto que almacena los datos del formulario.
   * 
   * @type {{ [key: string]: any }}
   * @memberof AgLoginFormComponent
   */
  form_object: { [key: string]: any } = {};

  /**
   * Datos de autenticación recibidos del servidor.
   * 
   * @type {any}
   * @memberof AgLoginFormComponent
   */
  authData: any

  /**
   * Crea una instancia del componente.
   * 
   * @param {ApiService} apiService - Servicio para la comunicación con la API.
   * @param {Router} router - Servicio de enrutamiento para la navegación.
   * @memberof AgLoginFormComponent
   */
  constructor(private apiService: ApiService, private router:Router){ }

  /**
   * Inicializa el componente.
   * 
   * Recupera información de firma del servidor cuando el componente se inicializa.
   * 
   * @method
   * @memberof AgLoginFormComponent
   */
  ngOnInit(): void {
    this.apiService.get(`${environment.URL_BACKEND}/sign_info_requests/generate`).subscribe((res:any)=>{
      this.resData = res
    })
  }

  /**
   * Maneja el envío del formulario de inicio de sesión.
   * 
   * Envía los datos del formulario a la API para autenticación, guarda el token de acceso en el almacenamiento local
   * y navega a la página principal si la autenticación es exitosa.
   * 
   * @method
   * @memberof AgLoginFormComponent
   */
  onSubmit() {
    if (this.resData && this.resData.header_token) {
      const body = {
        "username": this.email?.value,
        "password": this.pass?.value,
        "hidden_token": this.resData.hidden_token
      };

      const headers = new HttpHeaders()
        .set('X-Header-Token', this.resData.header_token)
        .set('Content-Type', 'application/json')
        .set('accept', 'application/json')
        .set('X-CSRFToken', 'zkPYf9nT7BuYjgzykQgOHPF9tkI9PmZyZMup8jpartlf2vlTYVOzvcwwUbzwWFBY');

      this.apiService.post(`${environment.URL_BACKEND}/auth/`, body, { headers }).subscribe((resAuth: any) => {
        this.authData = resAuth;
        localStorage.setItem('access_token', JSON.stringify(this.authData.content.access_token))

        //Navega a la pagina principal
        this.router.navigate(['home'])
      });
    }
  }

  /**
   * Cambia al formulario de registro.
   * 
   * Emite un evento para indicar que el usuario desea cambiar al formulario de registro.
   * 
   * @method
   * @memberof AgLoginFormComponent
   */
  changeForm(){
    this.isNew.emit(true)
  }
}
