Componente para mostrar una tabla con funcionalidades de paginación.

import { Component, Input, Output, EventEmitter, forwardRef, OnInit } from '@angular/core';
import { NG_VALUE_ACCESSOR, FormsModule } from '@angular/forms';
import { ApiService } from '../../functions/api.service';
import { environment } from '../../../../environments/environment.development';
import { CommonModule } from '@angular/common';
import { ApiSearchInterface } from '../../interfaces/ApiSearchGroupInterface';
import { getBaseApiSearchInterface } from '../../functions/functions';
import { TableApiSchemaInterface } from '../../interfaces/TableAPISchemaInterface';
import { MatIconModule } from '@angular/material/icon';
import { CdkMenuModule } from '@angular/cdk/menu';

/**
 * Componente para mostrar una tabla con funcionalidades de paginación.
 * 
 * Este componente se encarga de gestionar la visualización de datos en una tabla con soporte para
 * paginación. Se integra con una API para obtener el esquema de la tabla y los datos, y proporciona
 * métodos para navegar entre las páginas de resultados. Utiliza íconos para facilitar la navegación.
 * 
 * @component
 */
@Component({
  selector: 'ag-table',
  standalone: true,
  providers: [
    {
      provide: NG_VALUE_ACCESSOR,
      useExisting: forwardRef(() => AgTableComponent),
      multi: true,
    },
  ],
  imports: [CommonModule, FormsModule, MatIconModule,CdkMenuModule ],
  templateUrl: './ag-table.component.html',
  styleUrl: './ag-table.component.css'
})

export class AgTableComponent implements OnInit{
  //INTERACTIONS//

  /**
 * Nombre de la clave de la tabla utilizada en el componente.
 * 
 * Esta propiedad define el nombre de la clave que se usa para identificar la tabla en el componente. 
 * La clave puede ser utilizada para referenciar o acceder a datos específicos de la tabla.
 * 
 * @type {string}
 * @memberof AgTableComponent
 */
  @Input() tableKeyName!: string;
   /**
   * Cantidad de elementos por página.
   * 
   * @property {number} pageSize
   * @memberof AgTableComponent
   */
  @Input() pageSize!: number;
  /**
   * Criterios de búsqueda para la API.
   * 
   * @property {ApiSearchInterface | undefined} api_search_criteria_group
   * @memberof AgTableComponent
   */
  @Input() api_search_criteria_group?: ApiSearchInterface;
  /**
   * Evento emitido cuando una fila es seleccionada.
   * 
   * @property {EventEmitter<any>} rowSelected
   * @memberof AgTableComponent
   */
  @Output() rowSelected = new EventEmitter<any>();
  /**
   * Evento emitido cuando se selecciona un ítem específico.
   * 
   * @property {EventEmitter<any>} onSelectedItem
   * @memberof AgTableComponent
   */
  @Output() onSelectedItem: EventEmitter<any> = new EventEmitter<any>

  // PARAMETERS //
  /**
   * Página actual en la paginación.
   * 
   * @property {number} currentPage
   * @memberof AgTableComponent
   */
  public currentPage = 1
  /**
   * Número total de páginas disponibles.
   * 
   * @property {number} pages
   * @memberof AgTableComponent
   */
  pages = 0;
  /**
   * Indicador de carga para deshabilitar interacciones durante la petición.
   * 
   * @property {boolean} loading
   * @memberof AgTableComponent
   */
  loading = false

  console = console
  /**
   * Esquema de la tabla obtenido de la API.
   * 
   * @property {TableApiSchemaInterface | undefined} tableSchema
   * @memberof AgTableComponent
   */
  tableSchema?: TableApiSchemaInterface
  /**
   * Contenido de la tabla obtenido de la API.
   * 
   * @property {any[] | undefined} tableContent
   * @memberof AgTableComponent
   */
  tableContent?: any[]

  /**
   * Constructor de la clase. Inyecta el servicio ApiService para interactuar con la API.
   * 
   * @param {ApiService} apiService - Servicio para realizar llamadas a la API.
   * @memberof AgTableComponent
   */
  public constructor(private apiService: ApiService) {
  }

  /**
   * Método de inicialización que se ejecuta al cargar el componente.
   * Realiza una llamada a la API para obtener el esquema de la tabla y los datos iniciales.
   * 
   * @memberof AgTableComponent
   */
  ngOnInit(): void {
    this.callSearch()
  }
  
  /**
   * Realiza la búsqueda de datos en la API basada en el esquema y criterios de búsqueda.
   * 
   * @memberof AgTableComponent
   */
  callSearch() {
    this.loading = true;
    this.apiService.get(environment.URL_TABLE_SCHEMA_API + "/" + this.tableKeyName).subscribe((data: any) => {
      this.tableSchema = data
      this.apiService.post(environment.URL_BACKEND + this.tableSchema?.api_path + "?page=" + this.currentPage, this.api_search_criteria_group ? this.api_search_criteria_group : getBaseApiSearchInterface()).subscribe((api_response: any) => {
        this.pages = api_response.pages
        this.tableContent = api_response.results
        this.loading = false;
      })
    })
  }

  /**
   * Navega a la primera página de resultados y realiza una nueva búsqueda.
   * 
   * @memberof AgTableComponent
   */
  goToFirst() {
    if(this.loading) return
    this.currentPage = 1
    this.callSearch()
  }

  /**
   * Navega a la página anterior y realiza una nueva búsqueda.
   * 
   * @memberof AgTableComponent
   */
  goToPrevious() {
    if(this.loading) return
    if (this.currentPage - 1 > 0) {
      this.currentPage--
      this.callSearch()
    }
  }

  /**
   * Navega a la siguiente página y realiza una nueva búsqueda.
   * 
   * @memberof AgTableComponent
   */
  goToNext() {
    if(this.loading) return
    if (this.currentPage + 1 <= this.pages) {
      this.currentPage++
      this.callSearch()
    }
  }

  /**
   * Navega a la última página de resultados y realiza una nueva búsqueda.
   * 
   * @memberof AgTableComponent
   */
  goToLast() {
    if(this.loading) return
    this.currentPage = this.pages
    this.callSearch()
  }

  scrollToTop(){
    document.getElementsByClassName('mat-drawer-content')[0].scroll({top:0,left:0,behavior: 'smooth'})
  }

}
