import { Component, ViewChild } from '@angular/core';
import { CommonModule } from '@angular/common';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { environment } from '../../../../../environments/environment.development';
import { AgSearchComponent } from '../../../widgets/ag-search/ag-search.component';
import { ApiSearchInterface } from '../../../interfaces/ApiSearchGroupInterface';
import { AgTableComponent } from '../../../widgets/ag-table/ag-table.component';
import { FormDefinitions } from '../../../constants';

/**
 * Componente para la gestión de relaciones.
 * 
 * Este componente maneja la visualización y administración de relaciones a través de un formulario y una tabla.
 * Incluye funcionalidades para actualizar la interfaz de búsqueda y cargar datos en el formulario.
 * 
 * @component
 * @memberof AgRelationsComponent
 */
@Component({
  selector: 'app-ag-relations',
  standalone: true,
  imports: [CommonModule, AgFormComponent, AgSearchComponent, AgTableComponent],
  templateUrl: './ag-relations.component.html',
  styleUrl: './ag-relations.component.css'
})
export class AgRelationsComponent {
  
  /**
   * Entorno de la aplicación.
   * 
   * Proporciona acceso a las variables y configuraciones del entorno de la aplicación.
   * 
   * @type {any}
   * @memberof AgRelationsComponent
   */
  environment = environment

  /**
   * Definiciones de formulario.
   * 
   * Proporciona acceso a las constantes relacionadas con la definición de formularios.
   * 
   * @type {FormDefinitions}
   * @memberof AgRelationsComponent
   */
  FormDefinitions = FormDefinitions

  /**
   * Grupo de búsqueda de API para la tabla de relaciones.
   * 
   * Define la interfaz de búsqueda para la tabla de relaciones. Puede ser `undefined` si no se ha configurado.
   * 
   * @type {ApiSearchInterface | undefined}
   * @memberof AgRelationsComponent
   */
  tabla_relations_api_search_group?: ApiSearchInterface

  /**
   * Referencia al componente de tabla de relaciones.
   * 
   * Utilizado para interactuar con la tabla de relaciones en la vista.
   * 
   * @type {AgTableComponent | undefined}
   * @memberof AgRelationsComponent
   */
  @ViewChild('tabla_relations') tabla_relations?: AgTableComponent

  /**
   * Referencia al componente de formulario de relaciones.
   * 
   * Utilizado para interactuar con el formulario de relaciones en la vista.
   * 
   * @type {AgFormComponent | undefined}
   * @memberof AgRelationsComponent
   */
  @ViewChild('FORM_TABLE_RELATIONS') FORM_TABLE_RELATIONS?: AgFormComponent

  /**
   * Actualiza la interfaz de búsqueda de la tabla de relaciones.
   * 
   * Configura el grupo de búsqueda para la tabla y reinicia la página actual a 1 antes de realizar una búsqueda.
   * 
   * @param {ApiSearchInterface} apiSearchInterface - El grupo de búsqueda de API para actualizar la tabla.
   * @memberof AgRelationsComponent
   */
  update_tabla_relations_search_interface(apiSearchInterface: ApiSearchInterface){ 
    this.tabla_relations_api_search_group = apiSearchInterface
    this.tabla_relations!.currentPage = 1
    this.tabla_relations?.callSearch()
  }

  /**
   * Variable de estado para mostrar los resultados de la tabla de relaciones.
   * 
   * Indica si los resultados deben mostrarse en la tabla de relaciones.
   * 
   * @type {boolean}
   * @memberof AgRelationsComponent
   */
  mostrarResultadosTablaRelations = true

  /**
   * Carga datos en el formulario después de un evento.
   * 
   * Activa el formulario y asigna el objeto de formulario con los datos del evento después de un retraso de 250 ms.
   * 
   * @param {any} event - Los datos del evento que se cargarán en el formulario.
   * @memberof AgRelationsComponent
   */
  loadIntoForm(event: any){
    this.reveal_form = true
    setTimeout(() => {
      this.FORM_TABLE_RELATIONS!.form_object = event
    }, 250);
  }

  /**
   * Bandera para revelar el formulario.
   * 
   * Controla la visibilidad del formulario en la vista.
   * 
   * @type {boolean}
   * @memberof AgRelationsComponent
   */
  reveal_form = false

}
