import { Component, Input, Output, EventEmitter } from '@angular/core';

/**
 * Componente de Angular para mostrar un temporizador con capacidad de pausa y reanudaci√≥n.
 * 
 * Este componente gestiona un temporizador que cuenta regresivamente desde un tiempo definido en minutos.
 * Incluye la funcionalidad para pausar y reanudar el temporizador, y emite un evento cuando el temporizador finaliza.
 * 
 * @component
 */
@Component({
  selector: 'ag-timer',
  standalone: true,
  templateUrl: './ag-timer.component.html',
  styleUrls: ['./ag-timer.component.css']
})
export class AgTimerComponent {

  /**
   * Tiempo total del temporizador en minutos.
   * 
   * @property {number} endTime
   * @memberof AgTimerComponent
   */
  @Input() endTime!: number;
  /**
   * Evento emitido cuando el temporizador finaliza.
   * 
   * @property {EventEmitter<void>} timerFinished
   * @memberof AgTimerComponent
   */
  @Output() timerFinished = new EventEmitter<void>();

  /**
   * Tiempo restante en segundos.
   * 
   * @property {number} timeLeft
   * @memberof AgTimerComponent
   */
  timeLeft: number = 0;
  /**
   * Identificador del intervalo de temporizador.
   * 
   * @property {any} intervalId
   * @memberof AgTimerComponent
   */
  intervalId: any;

  /**
   * Indicador de si el temporizador est√° pausado.
   * 
   * @property {boolean} isPaused
   * @memberof AgTimerComponent
   */
  isPaused = false;

  /**
   * M√©todo de inicializaci√≥n que se ejecuta al cargar el componente.
   * Inicia el temporizador.
   * 
   * @memberof AgTimerComponent
   */
  ngOnInit(): void {
    this.startTimer();
  }

  /**
   * Inicia el temporizador.
   * Configura el intervalo para decrementar el tiempo restante cada segundo.
   * Cuando el tiempo restante llega a cero, emite un evento y detiene el temporizador.
   * 
   * @memberof AgTimerComponent
   */
  startTimer(): void {
    this.timeLeft = this.endTime * 60;
    this.intervalId = setInterval(() => {
      if (!this.isPaused) {
        this.timeLeft--;
        if (this.timeLeft === 0) {
          this.timerFinished.emit();
          clearInterval(this.intervalId);
        }
      }
    }, 1000);
  }

  /**
   * Formatea el tiempo en segundos a una cadena de formato HH:MM:SS.
   * 
   * @param {number} seconds - Tiempo en segundos.
   * @returns {string} - Tiempo formateado como HH:MM:SS.
   * @memberof AgTimerComponent
   */
  formatTime(seconds: number): string {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    const remainingSeconds = seconds % 60;
    return `${this.pad(hours)}:${this.pad(minutes)}:${this.pad(remainingSeconds)}`;
  }

  /**
   * A√±ade un cero a la izquierda si el n√∫mero es menor de 10.
   * 
   * @param {number} num - N√∫mero a formatear.
   * @returns {string} - N√∫mero formateado con un cero a la izquierda si es necesario.
   * @memberof AgTimerComponent
   */
  private pad(num: number): string {
    return num < 10 ? `0${num}` : `${num}`;
  }

  /**
   * Alterna el estado de pausa del temporizador.
   * 
   * @memberof AgTimerComponent
   */
  togglePause(): void {
    this.isPaused = !this.isPaused;
  }
}
