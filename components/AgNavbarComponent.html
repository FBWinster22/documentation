Description

Componente de barra de navegación para la aplicación.

El componente AgNavbarComponent proporciona una barra de navegación con enlaces dinámicos que se obtienen de un servicio API. Incluye métodos para alternar la visibilidad del menú de navegación y para buscar submenús basados en la ruta actual del enrutador.

import { Component, Input, OnInit } from '@angular/core';
import { MatSidenavModule } from '@angular/material/sidenav'
import { Router, RouterLink, RouterLinkActive } from '@angular/router'; 
import { MatIconModule } from '@angular/material/icon'
import { NavMenusInterface } from '../../interfaces/APIResponseInterface';
import { CommonModule } from '@angular/common';
import { MatExpansionModule } from '@angular/material/expansion';
import { ApiService } from '../../functions/api.service';
import { AgSubmenusComponent } from "../ag-submenus/ag-submenus.component";

/**
 * Componente de barra de navegación para la aplicación.
 * 
 * El componente `AgNavbarComponent` proporciona una barra de navegación con enlaces dinámicos que se obtienen de un servicio API. Incluye métodos para alternar la visibilidad del menú de navegación y para buscar submenús basados en la ruta actual del enrutador.
 * 
 * @component
 * @selector ag-navbar
 * @templateUrl ./ag-navbar.component.html
 * @styleUrl ./ag-navbar.component.css
 * @imports [MatSidenavModule, RouterLink, RouterLinkActive, MatIconModule, CommonModule, MatExpansionModule, AgSubmenusComponent]
 */
@Component({
    selector: 'ag-navbar',
    standalone: true,
    templateUrl: './ag-navbar.component.html',
    styleUrl: './ag-navbar.component.css',
    imports: [MatSidenavModule, RouterLink, RouterLinkActive, MatIconModule, CommonModule, MatExpansionModule, AgSubmenusComponent]
})
export class AgNavbarComponent implements OnInit{

  /**
   * URL del endpoint para obtener los enlaces de navegación.
   * 
   * Esta propiedad se utiliza para especificar la URL del servicio API que proporciona los datos para los enlaces de navegación. Debe ser una URL válida que apunte al recurso de enlaces de navegación.
   * 
   * @memberof AgNavbarComponent
   * @type {string}
   */
  @Input() linksUrl!:string;
  /**
   * Estado de apertura del menú de navegación.
   * 
   * Indica si el menú de navegación está abierto (`true`) o cerrado (`false`).
   * 
   * @memberof AgNavbarComponent
   * @type {boolean}
   */
  isNavOpen : boolean = false
  /**
   * Lista de menús de navegación obtenidos del servicio API.
   * 
   * Esta propiedad almacena los datos de los menús de navegación obtenidos a través del servicio API. Cada elemento de la lista representa un menú que se mostrará en la barra de navegación.
   * 
   * @memberof AgNavbarComponent
   * @type {NavMenusInterface[]}
   */
  navRes:NavMenusInterface[] = []
  /**
   * Menú actual basado en la ruta activa.
   * 
   * Esta propiedad almacena el menú actual que corresponde a la ruta activa en el enrutador. Se utiliza para resaltar el menú activo en la barra de navegación.
   * 
   * @memberof AgNavbarComponent
   * @type {NavMenusInterface[]}
   */
  currentMenu:NavMenusInterface[] = []
  /**
   * Submenús del menú actual.
   * 
   * Esta propiedad almacena los submenús asociados al menú actual. Se actualiza cuando se selecciona un menú en la barra de navegación.
   * 
   * @memberof AgNavbarComponent
   * @type {NavMenusInterface[]}
   */
  currentSubmenus:NavMenusInterface[] = []

  /**
   * Crea una instancia del componente `AgNavbarComponent`.
   * 
   * @param {ApiService} apiService - El servicio API utilizado para obtener los enlaces de navegación.
   * @param {Router} router - El enrutador utilizado para obtener la ruta actual.
   * @memberof AgNavbarComponent
   */
  constructor(private apiService: ApiService, private router:Router) { }

  /**
   * Inicializa el componente.
   * 
   * Este método se ejecuta cuando el componente se inicializa. Realiza una solicitud al servicio API para obtener los datos de los enlaces de navegación y los asigna a la propiedad `navRes`.
   * 
   * @memberof AgNavbarComponent
   * @method ngOnInit
   * @returns {void}
   */
  ngOnInit(): void {
    this.apiService.get(`${this.linksUrl}`).subscribe((res:any)=>{
      this.navRes = res
    })
  }

  /**
   * Alterna el estado de apertura del menú de navegación.
   * 
   * Este método cambia el valor de la propiedad `isNavOpen` para abrir o cerrar el menú de navegación.
   * 
   * @memberof AgNavbarComponent
   * @method toggleNav
   * @returns {void}
   */
  toggleNav(){
    this.isNavOpen = !this.isNavOpen
  }

  /**
   * Busca los submenús basados en la ruta activa.
   * 
   * Este método obtiene la ruta principal de la URL activa y encuentra el menú correspondiente en la propiedad `navRes`. Actualiza las propiedades `currentMenu` y `currentSubmenus` con el menú y los submenús encontrados.
   * 
   * @memberof AgNavbarComponent
   * @method buscarSubmenus
   * @returns {void}
   */
  buscarSubmenus() { //Nombre de método inconsistente
    /*Nombre de variable inconsistente ---> */const rutaPrincipal = this.router.url.split('/').filter(parte => parte !== '')[0] ?? ''; // Divide en partes la ruta completa y filtra partes vacías
    if (!rutaPrincipal) return;
    this.currentMenu = []; // Resetea los menús actuales
    this.currentSubmenus = []; // Resetea los submenús actuales
    /*Nombre de variable inconsistente ---> */const menuEncontrado = this.navRes.find(menu => menu.directorio === `/${rutaPrincipal}`); // Encuentra el menú que coincide con la ruta principal
    if (menuEncontrado) {
      this.currentMenu.push(menuEncontrado);
      this.currentSubmenus = menuEncontrado.submenus;
      console.log(menuEncontrado);
    }
  }
}
