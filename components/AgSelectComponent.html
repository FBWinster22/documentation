Description

Componente select personalizado que permite seleccionar opciones desde un conjunto predefinido o realizar una búsqueda dinámica en una API.

Este componente es compatible con formularios reactivos a través de la implementación de ControlValueAccessor. Permite la integración con APIs para obtener opciones de selección, manejando tanto la lógica de búsqueda como la visualización de las opciones.

import { CommonModule } from '@angular/common';
import { Component, EventEmitter, Input, OnInit, Output, forwardRef } from '@angular/core';
import { ApiSearchInterface } from '../../interfaces/ApiSearchGroupInterface';
import { ApiService } from '../../functions/api.service';
import { environment } from '../../../../environments/environment.development';
import { getBaseApiSearchInterface, generateFormattedOptionsString } from '../../functions/functions';
import { NG_VALUE_ACCESSOR } from '@angular/forms';

/**
 * Componente `select` personalizado que permite seleccionar opciones
 * desde un conjunto predefinido o realizar una búsqueda dinámica en una API.
 * 
 * Este componente es compatible con formularios reactivos a través de la implementación
 * de `ControlValueAccessor`. Permite la integración con APIs para obtener opciones de
 * selección, manejando tanto la lógica de búsqueda como la visualización de las opciones.
 * 
 * @component
 */
@Component({
  selector: 'ag-select',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './ag-select.component.html',
  styleUrl: './ag-select.component.css',
  providers: [
    {
      provide: NG_VALUE_ACCESSOR,
      useExisting: forwardRef(() => AgSelectComponent),
      multi: true,
    },
  ],
})
export class AgSelectComponent implements OnInit {

  /**
   * Opciones para el componente select, se espera un array de objetos con `value` y `text`.
   * 
   * @type {any[]}
   * @memberof AgSelectComponent
   */
  @Input() options: any[] = [];
  /**
   * Etiqueta para el componente select.
   * 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() label?: string = "";

  /**
   * Valor actualmente seleccionado en el componente select.
   * 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() value?: string = "";
  /**
   * Indica si el valor es válido.
   * 
   * @type {boolean}
   * @memberof AgSelectComponent
   */
  @Input() isValid?: boolean;

  /**
   * Nombre de la tabla en la base de datos utilizada para obtener datos desde la API.
   * 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() table_name?: string = "";

  /**
   * Ruta de la API desde la cual obtener los datos.
   * 
   * Si viene el campo de la `API`, **USAR ESE, NO EL API SEARCH** 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() api_path?: string = "";

  /**
   * Cuerpo de la solicitud para la API.
   * 
   * @type {object}
   * @memberof AgSelectComponent
   */
  @Input() request_body?: object

  /**
   * Valor para extraer de los resultados de la API.
   * 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() select_value?: string = "";
  /**
   * Texto para mostrar en las opciones extraídas de los resultados de la API.
   * 
   * @type {string}
   * @memberof AgSelectComponent
   */
  @Input() select_text?: string = "";
  /**
   * Grupo de criterios de búsqueda para la API.
   * 
   * @type {ApiSearchInterface}
   * @memberof AgSelectComponent
   * 
   * @Nota  Un select puede desplegar sus opciones por un set definido de pares {"value":"text"}, o por un par desde API.
   */
  @Input() api_search_criteria_group?: ApiSearchInterface;

  /**
   * Emite un evento cuando se selecciona una opción.
   * 
   * @type {EventEmitter<any>}
   * @memberof AgSelectComponent
   */
  @Output() selectionChange = new EventEmitter<any>();
  /**
   * Emite un evento cuando el valor cambia.
   * 
   * @type {EventEmitter<any>}
   * @memberof AgSelectComponent
   */
  @Output() valueChange = new EventEmitter<any>();

  /**
   * Opción actualmente seleccionada.
   * 
   * @type {any}
   * @private
   * @memberof AgSelectComponent
   */
  selectedOption: any = null;

  /**
   * Valor interno utilizado para el ControlValueAccessor.
   * 
   * @type {any}
   * @private
   * @memberof AgSelectComponent
   */
  private innerValue: any = '';

  /**
   * Constructor del componente. Inicializa el servicio de API.
   * 
   * @param {ApiService} apiService - Servicio para realizar solicitudes a la API.
   * @memberof AgSelectComponent
   */
  public constructor(private apiService: ApiService) {
  }

  /**
   * Método de ciclo de vida de Angular. Se llama cuando el componente se inicializa.
   * 
   * Llama al método `populate` para llenar las opciones del select.
   * 
   * @memberof AgSelectComponent
   * @method ngOnInit
   * @returns {void}
   */
  ngOnInit(): void {
    this.populate()
  }

  /**
   * Llena las opciones del select a partir de la API.
   * 
   * Este método realiza una solicitud POST a la API utilizando `api_path` y `request_body` para obtener las opciones. Luego, las opciones se formatean y se asignan al array `options`.
   * 
   * @memberof AgSelectComponent
   * @method populate
   * @returns {void}
   */
  populate() {

    if (this.api_path == undefined || this.api_path == '') {
      this.api_path = "/api/search/" + this.table_name + "/100/"
    }
    this.options = []
    this.apiService.post(environment.URL_BACKEND + this.api_path + "?page=1", this.request_body != undefined ? this.request_body : this.api_search_criteria_group ? this.api_search_criteria_group : getBaseApiSearchInterface()).subscribe((data: any) => {

      if (this.options.length == 0) {
        let options_template: any[] = []
        data.results.forEach((element: any) => {
          options_template.push({
            value: element[this.select_value!],
            text: generateFormattedOptionsString(this.select_text!, element)
          })
        });

        this.options = options_template
      }
    })
  }

  /**
   * Maneja la selección de una opción en el componente select.
   * 
   * Este método actualiza `selectedOption` con el valor seleccionado y emite el evento `selectionChange`.
   * 
   * @param {any} option - Opción seleccionada.
   * @memberof AgSelectComponent
   * @method onSelect
   * @returns {void}
   */
  onSelect(option: any): void {
    this.selectedOption = option.target.value;
    this.selectionChange.emit(this.selectedOption);
  }

  /**
   * Función que se ejecuta cuando el valor del componente cambia.
   * 
   * Esta función se registra a través del método `registerOnChange` como parte de la implementación de `ControlValueAccessor`.
   * Inicialmente es una función vacía que se sobrescribe con una función personalizada al registrarse.
   * 
   * @type {any}
   * @memberof AgSelectComponent
   */
  onChange: any = () => { };
  /**
   * Función que se ejecuta cuando el componente es tocado.
   * 
   * Esta función se registra a través del método `registerOnTouched` como parte de la implementación de `ControlValueAccessor`.
   * Inicialmente es una función vacía que se sobrescribe con una función personalizada al registrarse.
   * 
   * @type {any}
   * @memberof AgSelectComponent
   */
  onTouched: any = () => { };

  /**
   * Maneja el evento de entrada del componente select.
   * 
   * Actualiza el valor del select basado en el valor del input recibido.
   * 
   * @param {Event} event - Evento de entrada del componente select.
   * @memberof AgSelectComponent
   * @method onInput
   * @returns {void}
   */
  onInput(event: Event): void {
    const inputElement = event.target as HTMLInputElement;
    if (inputElement) {
      this.value = inputElement.value;
    }
  }

  /**
   * Maneja el evento de desenfoque del componente select.
   * 
   * Llama al método `onTouched` para indicar que el componente ha sido tocado.
   * 
   * @memberof AgSelectComponent
   * @method onBlur
   * @returns {void}
   */
  onBlur() {
    this.onTouched();
  }

  /**
   * Implementa el método `writeValue` del interfaz `ControlValueAccessor`.
   * 
   * Actualiza el valor interno del componente con el valor proporcionado.
   * 
   * @param {any} value - Valor que se debe escribir en el componente.
   * @memberof AgSelectComponent
   * @method writeValue
   * @returns {void}
   */
  writeValue(value: any) {
    this.innerValue = value;
  }

  /**
   * Implementa el método `registerOnChange` del interfaz `ControlValueAccessor`.
   * 
   * Registra una función que se llamará cuando el valor del componente cambie.
   * 
   * @param {any} fn - Función a registrar para cambios de valor.
   * @memberof AgSelectComponent
   * @method registerOnChange
   * @returns {void}
   */
  registerOnChange(fn: any) {
    this.onChange = fn;
  }

  /**
   * Implementa el método `registerOnTouched` del interfaz `ControlValueAccessor`.
   * 
   * Registra una función que se llamará cuando el componente sea tocado.
   * 
   * @param {any} fn - Función a registrar para el evento de toque.
   * @memberof AgSelectComponent
   * @method registerOnTouched
   * @returns {void}
   */
  registerOnTouched(fn: any) {
    this.onTouched = fn;
  }

  /**
   * Maneja el cambio de valor del componente select.
   * 
   * Actualiza el valor del componente y emite el evento `valueChange` con el nuevo valor.
   * 
   * @param {any} option - Nuevo valor seleccionado.
   * @memberof AgSelectComponent
   * @method onValueChanged
   * @returns {void}
   */
  onValueChanged(option: any) {
    this.value = option;
    // Emitir evento ngModelChange
    this.valueChange.emit(option);
  }

}
