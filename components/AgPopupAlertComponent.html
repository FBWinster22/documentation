Componente de alerta emergente.

import { Component, EventEmitter, input, Input, OnChanges, Output, SimpleChanges } from '@angular/core';
import { CommonModule } from '@angular/common';

/**
 * Componente de alerta emergente.
 * 
 * El componente `AgPopupAlertComponent` muestra una alerta emergente con un mensaje y un color de fondo configurable. La alerta se desvanecerá después de un breve periodo y notificará al componente padre cuando se cierre.
 * 
 * @component
 * @selector ag-popup-alert
 * @templateUrl ./ag-popup-alert.component.html
 * @styleUrl ./ag-popup-alert.component.css
 * @imports [CommonModule]
 */
@Component({
  selector: 'ag-popup-alert',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './ag-popup-alert.component.html',
  styleUrl: './ag-popup-alert.component.css'
})
export class AgPopupAlertComponent {

  /**
   * Estado de la alerta.
   * 
   * Esta propiedad indica si la alerta debe mostrarse (`true`) o no (`false`). Cuando se establece en `true`, se activa la visualización de la alerta.
   * 
   * @memberof AgPopupAlertComponent
   * @type {boolean}
   */
  @Input() isAlert?: boolean;
  /**
   * Mensaje de la alerta.
   * 
   * Esta propiedad define el mensaje que se mostrará en la alerta emergente. Si no se proporciona, la alerta no tendrá mensaje.
   * 
   * @memberof AgPopupAlertComponent
   * @type {string}
   */
  @Input() message?: string;
  /**
   * Color de fondo de la alerta.
   * 
   * Esta propiedad define el color de fondo de la alerta. El valor predeterminado es `#ffffff` (blanco).
   * 
   * @memberof AgPopupAlertComponent
   * @type {string}
   * @default '#ffffff'
   */
  @Input() backgroundColor:string = '#ffffff'
  /**
   * Evento emitido cuando la alerta se cierra.
   * 
   * Este evento se emite cuando la alerta se ha desvanecido completamente. Se utiliza para notificar al componente padre que la alerta ha sido cerrada.
   * 
   * @memberof AgPopupAlertComponent
   * @type {EventEmitter<void>}
   */
  @Output() alertClosed = new EventEmitter<void>(); // Para notificar al componente padre
  /**
   * Estado de desvanecimiento de la alerta.
   * 
   * Esta propiedad controla el estado de desvanecimiento de la alerta. Se utiliza para aplicar animaciones CSS de desvanecimiento.
   * 
   * @memberof AgPopupAlertComponent
   * @type {boolean}
   */
  fadeOut = false;

  /**
   * Detecta los cambios en las propiedades de entrada.
   * 
   * Este método se llama cuando cambian las propiedades de entrada del componente. Si la propiedad `isAlert` es `true`, se muestra la alerta.
   * 
   * @memberof AgPopupAlertComponent
   * @method ngOnChanges
   * @param {SimpleChanges} changes - Los cambios en las propiedades de entrada.
   * @returns {void}
   */
  ngOnChanges() {
    console.log(this.backgroundColor)
    if (this.isAlert) {
      this.showAlert();
    }
  }

  /**
   * Muestra la alerta y maneja la animación de desvanecimiento.
   * 
   * Este método inicia la animación de la alerta. La alerta se muestra durante un periodo y luego se desvanecerá. Después de completar la animación, se emite el evento `alertClosed`.
   * 
   * @memberof AgPopupAlertComponent
   * @method showAlert
   * @returns {void}
   */
  showAlert() {
    this.fadeOut = false; // Reinicia el estado de fadeOut
    setTimeout(() => {
      this.fadeOut = true;
      setTimeout(() => {
        this.alertClosed.emit();
      }, 500); // Tiempo para completar la animación fadeOut
    }, 2000); // Tiempo de espera antes de iniciar fadeOut
  }

}

  <div class="modal-overlay">

      <div class="alert" [ngStyle]="{'background-color': backgroundColor}">

        <h1>{{ message }}</h1>
      </div>
    </div>
