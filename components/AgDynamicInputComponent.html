Description

El componente AgDynamicInputComponent es un componente dinámico que gestiona diferentes tipos de entradas, incluyendo la apertura de modales de búsqueda, selección de opciones, y visualización de tablas de resultados. Es altamente configurable y puede manejar eventos como la selección de ítems, cambios en los archivos, y la actualización de interfaces de búsqueda.

import { CommonModule } from '@angular/common';
import { Component, Input, Output, EventEmitter, ViewChild, OnInit } from '@angular/core';
import { AgSelectComponent } from '../ag-select/ag-select.component';
import { AgInputComponent } from '../ag-input/ag-input.component';
import { AgModalComponent } from '../ag-modal/ag-modal.component';
import { AgSearchComponent } from '../ag-search/ag-search.component';
import { AgTableComponent } from '../ag-table/ag-table.component';
import { FormsModule } from '@angular/forms';
import { NgxColorsModule } from 'ngx-colors';
import { MatIconModule } from '@angular/material/icon';
import { AgFormComponent } from "../ag-form/ag-form.component";
import { environment } from '../../../../environments/environment.development';
import { AgChildFormComponent } from '../ag-child-form/ag-child-form.component';
import { BaseDynamicInputComponent } from '../BaseDynamicInputComponent';

/**
 * El componente `AgDynamicInputComponent` es un componente dinámico que gestiona diferentes tipos de entradas,
 * incluyendo la apertura de modales de búsqueda, selección de opciones, y visualización de tablas de resultados.
 * Es altamente configurable y puede manejar eventos como la selección de ítems, cambios en los archivos, 
 * y la actualización de interfaces de búsqueda.
 * @memberof AgDynamicInputComponent
 */
@Component({
  imports: [AgChildFormComponent, CommonModule, MatIconModule, AgSelectComponent, AgInputComponent, AgModalComponent, AgSearchComponent, AgTableComponent, FormsModule, NgxColorsModule, AgFormComponent],
  standalone: true,
  selector: 'ag-dynamic-input',
  templateUrl: './ag-dynamic-input.component.html',
  styleUrls: ['./ag-dynamic-input.component.css'],

})
export class AgDynamicInputComponent extends BaseDynamicInputComponent {

  environment = environment
  json_child_form_data?:any
  reveal!:boolean

  @ViewChild('FORM_OPTIONS_CONFIG') FORM_OPTIONS_CONFIG?: AgFormComponent
  
  @Output() json_child_form:EventEmitter<any> = new EventEmitter<any>();

  handleModalValue(event: any) {
    this.json_child_form_data = event
    this.json_child_form.emit(this.json_child_form_data)
  }
}
