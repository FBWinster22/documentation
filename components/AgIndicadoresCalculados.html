Referencia al componente de formulario para indicadores calculados.

Utilizado para interactuar con el formulario de indicadores calculados. Esta referencia se establece mediante el decorador @ViewChild y permite acceder a los métodos y propiedades del componente AgFormComponent.

 
import { Component, ViewChild } from '@angular/core';
import { AgSearchComponent } from '../../../widgets/ag-search/ag-search.component';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { CommonModule } from '@angular/common';
import { environment } from '../../../../../environments/environment.development';

/**
 * Componente para la visualización y manejo de indicadores calculados.
 * 
 * Este componente utiliza los widgets `AgSearchComponent` y `AgFormComponent` para
 * proporcionar funcionalidades de búsqueda y formularios dentro de la aplicación.
 * 
 * @component
 * @memberof TemplateComponent
 */
@Component({
  selector: 'app-template',
  standalone: true,
  imports: [AgSearchComponent, AgFormComponent, CommonModule],
  templateUrl: './ag-indicadores-calculados.component.html',
  styleUrl: './ag-indicadores-calculados.component.css'
})
export class AgIndicadoresCalculados {
/**
   * Referencia al componente de formulario para indicadores calculados.
   * 
   * Utilizado para interactuar con el formulario de indicadores calculados. Esta referencia
   * se establece mediante el decorador `@ViewChild` y permite acceder a los métodos y
   * propiedades del componente `AgFormComponent`.
   * 
   * @type {AgFormComponent}
   * @memberof TemplateComponent
   */
  
  @ViewChild('FORM_TABLE_INDICADORES_CALCULADOS') FORM_TABLE_INDICADORES_CALCULADOS !: AgFormComponent

  environment = environment
}
