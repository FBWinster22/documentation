import { Component, EventEmitter, Output } from '@angular/core';
import { AgInputComponent } from '../../../../../widgets/ag-input/ag-input.component';
import { AgSocialMediaComponent } from '../../../../../widgets/ag-social-media/ag-social-media.component';

/**
 * Componente para el formulario de registro.
 * 
 * Este componente proporciona una interfaz para que los usuarios ingresen su información de registro,
 * incluyendo campos para datos de usuario y medios sociales.
 * 
 * @component
 * @memberof AgRegisterFormComponent
 */
@Component({
  selector: 'ag-register-form',
  standalone: true,
  imports: [AgInputComponent, AgSocialMediaComponent],
  templateUrl: './ag-form-register.component.html',
  styleUrl: './ag-form-register.component.css'
})
export class AgRegisterFormComponent {

  /**
   * Emisor de eventos que indica si se debe cambiar al formulario de inicio de sesión.
   * 
   * @type {EventEmitter<any>}
   * @memberof AgRegisterFormComponent
   */
  @Output() isNew: EventEmitter<any> = new EventEmitter<any>()

  /**
   * Cambia al formulario de inicio de sesión.
   * 
   * Emite un evento para indicar que el usuario desea cambiar al formulario de inicio de sesión.
   * 
   * @method
   * @memberof AgRegisterFormComponent
   */
  changeForm(){
    this.isNew.emit(false)
  }
}
