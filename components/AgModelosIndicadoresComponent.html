import { Component, ViewChild } from '@angular/core';
import { CommonModule } from '@angular/common';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { environment } from '../../../../../environments/environment.development';
import { AgSearchComponent } from '../../../widgets/ag-search/ag-search.component';
import { ApiSearchInterface } from '../../../interfaces/ApiSearchGroupInterface';
import { AgTableComponent } from '../../../widgets/ag-table/ag-table.component';

/**
 * Componente para la gestión de modelos e indicadores.
 * 
 * Este componente proporciona funcionalidades para la visualización y manipulación de datos relacionados
 * con modelos y máquinas, incluyendo formularios, búsquedas y tablas.
 * 
 * @component
 * @memberof AgModelosIndicadoresComponent
 */
@Component({
  selector: 'app-ag-modelos-indicadores',
  standalone: true,
  imports: [AgFormComponent, AgSearchComponent, AgTableComponent, CommonModule],
  templateUrl: './ag-modelos-indicadores.component.html',
  styleUrl: './ag-modelos-indicadores.component.css'
})
export class AgModelosIndicadoresComponent {

  /**
   * Entorno de la aplicación.
   * 
   * Proporciona acceso a las variables y configuraciones del entorno de la aplicación.
   * 
   * @type {any}
   * @memberof AgModelosIndicadoresComponent
   */
  environment = environment

  /**
   * Referencia al componente de formulario para modelos y máquinas.
   * 
   * Utiliza `@ViewChild` para obtener una referencia al componente `AgFormComponent` definido en la plantilla.
   * 
   * @type {AgFormComponent | undefined}
   * @memberof AgModelosIndicadoresComponent
   */
  @ViewChild('FORM_MODELOS_MAQUINAS') FORM_MODELOS_MAQUINAS?: AgFormComponent
  
  /**
   * Referencia al componente de tabla para modelos y máquinas.
   * 
   * Utiliza `@ViewChild` para obtener una referencia al componente `AgTableComponent` definido en la plantilla.
   * 
   * @type {AgTableComponent | undefined}
   * @memberof AgModelosIndicadoresComponent
   */
  @ViewChild('tabla_modelos_maquinas') tabla_modelos_maquinas?: AgTableComponent

  /**
   * Configuración del grupo de búsqueda para la tabla de modelos y máquinas.
   * 
   * Almacena la interfaz de búsqueda que se utiliza para realizar consultas en la tabla.
   * 
   * @type {ApiSearchInterface | undefined}
   * @memberof AgModelosIndicadoresComponent
   */
  tabla_modelos_maquinas_api_search_group?: ApiSearchInterface

  /**
   * Actualiza la configuración de búsqueda de la tabla de modelos y máquinas.
   * 
   * Establece la interfaz de búsqueda y reinicia la página actual de la tabla, luego llama a la función de búsqueda.
   * 
   * @param {any} apiSearchInterface - Interfaz de búsqueda para la tabla.
   * @memberof AgModelosIndicadoresComponent
   */
  update_tabla_paquetesisg_search_interface(apiSearchInterface: any){

    this.tabla_modelos_maquinas_api_search_group = apiSearchInterface
    this.tabla_modelos_maquinas!.currentPage = 1
    this.tabla_modelos_maquinas?.callSearch()

  }

  /**
   * Bandera para mostrar o ocultar los resultados en la tabla de modelos y máquinas.
   * 
   * Inicialmente establecido en `false`. El nombre de esta propiedad es inconsistente y se recomienda revisar.
   * 
   * @type {boolean}
   * @memberof AgModelosIndicadoresComponent
   */
  mostrarResultadosTablaModelosMaquinas = false //Nombre inconsistente

  /**
   * Maneja la carga de datos en el formulario.
   * 
   * Muestra el formulario y asigna los datos del evento a la propiedad `form_object` del componente `AgFormComponent`.
   * 
   * @param {any} event - Datos recibidos que se asignarán al formulario.
   * @memberof AgModelosIndicadoresComponent
   */
  loadIntoForm(event: any){
    this.reveal_form = true
    setTimeout(() => {
      this.FORM_MODELOS_MAQUINAS!.form_object = event
    }, 250);
  }

  /**
   * Variable de estado para revelar u ocultar el formulario.
   * 
   * Inicialmente establecido en `false`. Controla la visibilidad del formulario basado en la lógica de la aplicación.
   * 
   * @type {boolean}
   * @memberof AgModelosIndicadoresComponent
   */
  reveal_form = false
}

<ag-form #FORM_TABLA_MODELOS_MAQUINAS [formDefinition]="'FORM_TABLA_MODELOS_MAQUINAS'" [endpoint]="environment.URL_BACKEND+'/modelos_maquinas'" [reveal]="true"></ag-form>
<ag-search [simplified]="true" [tableKeyName]="'tabla_modelos_maquinas'" [formRef]="FORM_TABLA_MODELOS_MAQUINAS"/>
