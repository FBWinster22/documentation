import { Component, Output, EventEmitter, HostListener } from '@angular/core';

/**
 * Componente de modal para la aplicación.
 * 
 * El componente `AgModalComponent` proporciona una ventana modal que puede ser utilizada para mostrar contenido adicional y permite cerrar el modal mediante un evento. El modal emite un evento `closeModal` cuando se cierra, que puede ser utilizado por otros componentes para manejar el cierre del modal.
 * 
 * @component
 * @selector ag-modal
 */
@Component({
  standalone: true,
  selector: 'ag-modal',
  templateUrl: './ag-modal.component.html',
  styleUrl: './ag-modal.component.css'
})
export class AgModalComponent {
  @HostListener('document:keydown.escape', ['$event'])
  handleEsc(event: KeyboardEvent) {
    this.close();
  }

  /**
   * Evento emitido cuando se cierra el modal.
   * 
   * Este evento se emite cuando se llama al método `close`. Los componentes que utilizan `AgModalComponent` pueden suscribirse a este evento para manejar el cierre del modal.
   * 
   * @memberof AgModalComponent
   * @type {EventEmitter<void>}
   */
  @Output() closeModal = new EventEmitter<void>();

  /**
   * Cierra el modal.
   * 
   * Este método emite el evento `closeModal`, lo que indica que el modal debe ser cerrado. Los componentes que escuchan este evento pueden realizar las acciones necesarias para ocultar el modal.
   * 
   * @memberof AgModalComponent
   * @method close
   * @returns {void}
   * @fires closeModal - Emite el evento `closeModal` para notificar que el modal ha sido cerrado.
   */
  close() {
    this.closeModal.emit();
  }
}
