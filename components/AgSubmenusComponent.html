import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { ApiService } from '../../functions/api.service'; // Asegúrate de importar correctamente tu servicio
import { environment } from '../../../../environments/environment.development';

/**
 * Componente que maneja la visualización y selección de submenús y sub-submenús.
 * Este componente se encarga de cargar los menús desde una API, y permite seleccionar submenús
 * para mostrar sus sub-submenús asociados. Es autónomo y se puede utilizar para construir
 * interfaces de navegación complejas.
 * 
 * @component
 */
@Component({
  selector: 'ag-submenus',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './ag-submenus.component.html',
  styleUrl: './ag-submenus.component.css'
})
export class AgSubmenusComponent implements OnInit {

  /**
   * Constructor de la clase. Inyecta el servicio ApiService para interactuar con la API.
   * 
   * @param {ApiService} apiService - Servicio para realizar llamadas a la API.
   * @memberof AgSubmenusComponent
   */
  constructor(private apiService: ApiService) { }

  /**
   * Arreglo que contiene los menús principales obtenidos de la API.
   * 
   * @property {any[]} listaMenus
   * @memberof AgSubmenusComponent
   */
  listaMenus: any[] = []; //Nombre inconsistente
  /**
   * ID del submenú seleccionado por el usuario.
   * 
   * @property {number | null} idSubmenuSeleccionado
   * @memberof AgSubmenusComponent
   */
  idSubmenuSeleccionado: number | null = null; //Nombre inconsistente
  /**
   * Arreglo que contiene los submenús anidados (subsubmenús) del submenú seleccionado.
   * 
   * **Inicializa como un array vacío**
   * @property {any[]} listaSubsubmenus
   * @memberof AgSubmenusComponent
   */
  listaSubsubmenus: any[] = []; //Nombre inconsistente
  /**
   * Objeto que representa el menú principal seleccionado.
   * 
   * @property {any} menu
   * @memberof AgSubmenusComponent
   */
  menu: any;
  /**
   * ID del submenú activo actualmente.
   * 
   * @property {number | null} activeSubmenuId
   * @memberof AgSubmenusComponent
   */
  activeSubmenuId: number | null = null;

  /**
   * Método que se ejecuta al inicializar el componente.
   * Obtiene la lista de menús principales desde la API y la almacena en `listaMenus`.
   * 
   * @memberof AgSubmenusComponent
   */
  ngOnInit() {
    this.apiService.get(environment.URL_BACKEND + "/nav_menus/").subscribe(
      (data: any) => {
        this.listaMenus = data;
        console.log(this.listaMenus); // Verifica que recibes los datos correctamente
      }
    );
  }

  /**
   * Busca y almacena los subsubmenús del submenú seleccionado.
   * 
   * @param {number} id - ID del submenú seleccionado.
   * @memberof AgSubmenusComponent
   */
  buscarSubsubmenus(id: number) { //Nombre inconsistente
    this.listaSubsubmenus = []; // Reiniciar la lista de subsubmenús antes de buscar
    for (let menu of this.listaMenus) {
      for (let submenu of menu.submenus) {
        if (submenu.id === id && submenu.submenus) {
          this.listaSubsubmenus.push(...submenu.submenus);
          break; // Detener la iteración si encontramos los submenús
        }
      }
    }
    this.activeSubmenuId = id;
    console.log('Subsubmenús encontrados:', this.listaSubsubmenus);
  }

  /**
   * Guarda el identificador del submenú seleccionado y busca los subsubmenús asociados.
   * 
   * @param {number} id - El identificador del submenú que se seleccionó.
   * @memberof AgSubmenusComponent
   */
  guardarIdSubmenu(id: number) { //Nombre inconsistente
    this.idSubmenuSeleccionado = id;
    console.log('ID del submenú seleccionado:', id);
    this.buscarSubsubmenus(id);
    // Aquí puedes realizar otras acciones con el ID, como enviarlo a un servicio o realizar una navegación, etc.
  }
}
