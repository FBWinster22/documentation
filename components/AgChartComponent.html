Description

Componente para visualizar gráficos utilizando Chart.js.

Este componente permite crear gráficos interactivos en diferentes tipos (barra, burbuja, dona, línea, pastel, área polar, radar y dispersión) a partir de los datos proporcionados. También incluye la funcionalidad para generar una imagen en formato PNG del gráfico.

Example :
<ag-chart [chartType]="'bar'" [chartData]="{ label: 'Sales', data: { 'January': 50, 'February': 75 } }"></ag-chart>

import { Component, Input, OnInit, ViewChild, ElementRef, AfterViewInit, EventEmitter, Output } from '@angular/core';
import Chart from 'chart.js/auto';
import html2canvas from 'html2canvas';

/**
 * Componente para visualizar gráficos utilizando Chart.js.
 * 
 * Este componente permite crear gráficos interactivos en diferentes tipos (barra, burbuja, dona, línea, pastel, área polar, radar y dispersión)
 * a partir de los datos proporcionados. También incluye la funcionalidad para generar una imagen en formato PNG del gráfico.
 * 
 * @example
 * <ag-chart [chartType]="'bar'" [chartData]="{ label: 'Sales', data: { 'January': 50, 'February': 75 } }"></ag-chart>
 */

@Component({
  selector: 'ag-chart',
  standalone: true,
  templateUrl: './ag-chart.component.html',
  styleUrls: ['./ag-chart.component.css']
})
export class AgChartComponent implements OnInit, AfterViewInit {

  /**
   * Tipo de gráfico a crear.
   * 
   * Puede ser uno de los siguientes: 'bar', 'bubble', 'doughnut', 'line', 'pie', 'polarArea', 'radar', 'scatter'.
   * 
   * @type {'bar' | 'bubble' | 'doughnut' | 'line' | 'pie' | 'polarArea' | 'radar' | 'scatter'}
   * @default 'bar'
   */
  @Input() chartType: 'bar' | 'bubble' | 'doughnut' | 'line' | 'pie' | 'polarArea' | 'radar' | 'scatter' = 'bar';
  
  /**
   * Datos del gráfico a mostrar.
   * 
   * Contiene una etiqueta para el gráfico y un conjunto de datos en formato clave-valor donde la clave es el nombre del eje X y el valor es el dato correspondiente.
   * 
   * @type {{ label: string; data: { [key: string]: number } }}
   * @default { label: '', data: {} }
   */
  @Input() chartData: { label: string; data: { [key: string]: number } } = { label: '', data: {} };

  /**
   * Referencia al elemento canvas donde se renderiza el gráfico.
   * 
   * @type {ElementRef<HTMLCanvasElement>}
   */
  @ViewChild('chart') chartRef!: ElementRef<HTMLCanvasElement>;

  /**
   * Instancia del gráfico creado con Chart.js.
   * 
   * @type {Chart<'bar' | 'bubble' | 'doughnut' | 'line' | 'pie' | 'polarArea' | 'radar' | 'scatter'> | undefined}
   */
  chartInstance: Chart<'bar' | 'bubble' | 'doughnut' | 'line' | 'pie' | 'polarArea' | 'radar' | 'scatter'> | undefined;
  
  /**
   * Imagen en formato PNG del gráfico generado con html2canvas.
   * 
   * @type {string | null}
   */
  chartImage: string | null = null;

  /**
   * Método del ciclo de vida de Angular que se ejecuta cuando el componente se inicializa.
   * 
   * @returns {void}
   */
  ngOnInit(): void {}

  /**
   * Método del ciclo de vida de Angular que se ejecuta después de que la vista del componente ha sido inicializada.
   * Crea el gráfico si no se ha creado aún.
   * 
   * @returns {void}
   */
  ngAfterViewInit(): void {
    if (this.chartRef && !this.chartInstance) {
      this.createChart();
    }
  }

  /**
   * Crea un gráfico utilizando Chart.js con el tipo y datos proporcionados.
   * Configura las opciones del gráfico, incluyendo el diseño, colores y etiquetas.
   * 
   * @returns {void}
   */
  createChart(): void {
    const ctx = this.chartRef.nativeElement.getContext('2d');
    if (!ctx) {
      console.error('Error obtaining 2D context from canvas element');
      return;
    }
  
    
    const labels = Object.keys(this.chartData.data);
    const data = Object.values(this.chartData.data);
  
    this.chartInstance = new Chart(ctx, {
      type: this.chartType,
      data: {
        labels: labels,
        datasets: [{
          label: this.chartData.label,
          data: data,
          backgroundColor: [
            '#367c2b',   // Verde oscuro
            '#e5e6e6',   // Gris claro
            '#ffde00',   // Amarillo brillante
            '#f28d35',   // Naranja cálido
            '#6e5b4d',   // Marrón oscuro
            '#007bff',   // Azul brillante
            '#ff6347',   // Rojo tomate
          ],
          borderColor: [
            '#367c2b',   // Verde oscuro
            '#e5e6e6',   // Gris claro
            '#ffde00',   // Amarillo brillante
            '#f28d35',   // Naranja cálido
            '#6e5b4d',   // Marrón oscuro
            '#007bff',   // Azul brillante
            '#ff6347',   // Rojo tomate
          ],
          borderWidth: 1
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          title: {
            display: true,
            text: this.chartData.label,
            font: {
              size: 16
            }
          },
          tooltip: {
            callbacks: {
              label: (context) => {
                return `${context.label}: ${context.raw}`;
              }
            }
          },
          legend: {
            display: true,
            labels: {
              font: {
                size: 14
              }
            }
          }
        },
      }
    });
  }

  /**
   * Genera una imagen en formato PNG del gráfico actual utilizando html2canvas.
   * Guarda la imagen generada en la propiedad `chartImage`.
   * 
   * @returns {void}
   */
  generateChartImage(): void {
    if (this.chartInstance) {
      html2canvas(this.chartRef.nativeElement).then(canvas => {
        this.chartImage = canvas.toDataURL('image/png');
      }).catch(err => {
        console.error('Error generating chart image:', err);
      });
    }
  }
}
