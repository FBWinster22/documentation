import { Component, ElementRef, HostListener, OnInit,  } from '@angular/core';
import { CommonModule } from '@angular/common';
import { AgContextMenuService } from '../../services/ag-context-menu.service';

@Component({
  selector: 'app-context-menu',
  templateUrl: 'ag-context-menu.component.html',
  styleUrls: ['./ag-context-menu.component.css']
})

export class AgContextMenuComponent implements OnInit {
  isVisible = false;
  position = { x: 0, y: 0 };

  constructor(
    private contextMenuService: AgContextMenuService,
    private el: ElementRef
  ) {}

  ngOnInit() {
    this.contextMenuService.isVisible$.subscribe(isVisible => this.isVisible = isVisible);
    this.contextMenuService.position$.subscribe(position => this.position = position);
  }

  @HostListener('document:click', ['$event'])
  onClickOutside(event: MouseEvent) {
    if (!this.el.nativeElement.contains(event.target)) {
      this.contextMenuService.hideMenu();
    }
  }

  onOptionClick(option: string) {
    console.log(option);
    this.contextMenuService.hideMenu();
  }
}
