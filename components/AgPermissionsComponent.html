import { Component } from '@angular/core';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { environment } from '../../../../../environments/environment.development';

/**
 * Componente para la gestión de permisos.
 * 
 * Este componente proporciona una interfaz para la administración de permisos dentro de la aplicación.
 * 
 * @component
 * @memberof AgPermissionsComponent
 */
@Component({
  selector: 'app-ag-permissions',
  standalone: true,
  imports: [AgFormComponent],
  templateUrl: './ag-permissions.component.html',
  styleUrl: './ag-permissions.component.css'
})
export class AgPermissionsComponent {

   /**
   * Entorno de la aplicación.
   * 
   * Proporciona acceso a las variables y configuraciones del entorno de la aplicación.
   * 
   * @type {any}
   * @memberof AgPermissionsComponent
   */
  environment = environment
}
