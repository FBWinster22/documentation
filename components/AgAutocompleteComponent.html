Description

El AgAutocompleteComponent es un componente autónomo en Angular diseñado para proporcionar funcionalidad de autocompletado en formularios. Este componente permite a los usuarios buscar y seleccionar opciones en una tabla específica mediante la integración con una API.

Utiliza el servicio ApiService para realizar solicitudes a la API y obtener los resultados correspondientes, que luego se filtran y se muestran al usuario a medida que escribe. El componente es altamente configurable, permitiendo especificar la tabla y la columna de búsqueda a través de propiedades de entrada (@Input).

import { HttpClient } from '@angular/common/http';
import { Component, OnInit, Input } from '@angular/core';
import { FormsModule } from '@angular/forms';
import { ApiService } from '../../functions/api.service';
import { CommonModule } from '@angular/common';
import { AgInputComponent } from '../ag-input/ag-input.component';
import { environment } from '../../../../environments/environment.development';

/**
 * @component AgAutocompleteComponent
 *
 * @description
 * El `AgAutocompleteComponent` es un componente autónomo en Angular diseñado para proporcionar
 * funcionalidad de autocompletado en formularios. Este componente permite a los usuarios buscar 
 * y seleccionar opciones en una tabla específica mediante la integración con una API.
 *
 * Utiliza el servicio `ApiService` para realizar solicitudes a la API y obtener los resultados
 * correspondientes, que luego se filtran y se muestran al usuario a medida que escribe. El 
 * componente es altamente configurable, permitiendo especificar la tabla y la columna de búsqueda
 * a través de propiedades de entrada (`@Input`).
 */
@Component({
  selector: 'ag-autocomplete',
  standalone: true,
  imports: [FormsModule, CommonModule, AgInputComponent],
  templateUrl: './ag-autocomplete.component.html',
  styleUrls: ['./ag-autocomplete.component.css']
})
export class AgAutocompleteComponent implements OnInit {
  /**
   * Parámetro para la tabla en la que se realizará la búsqueda.
   * 
   * @type {string}
   */
  @Input() tableParam: string = '';

   /**
   * Nombre de la columna que se utilizará para la búsqueda.
   * 
   * @type {string}
   */
  @Input() columnName: string = '';
  
  /**
   * Término de búsqueda actual introducido por el usuario.
   * 
   * @type {string}
   */
  searchTerm: string = '';

  /**
   *
   * Resultados obtenidos de la búsqueda en la API.
   * @type {any[]}
   * @memberof AgAutocompleteComponent
   */
  results: any[] = [];

  /**
   * Resultados filtrados basados en el término de búsqueda.
   * 
   * @type {any[]}
   */
  filteredResults: any[] = [];

  /**
   * Constructor del componente.
   * 
   * @param {HttpClient} http - El cliente HTTP para realizar solicitudes a la API.
   * @param {ApiService} apiService - Servicio para manejar las solicitudes a la API.
   */
  constructor(private http: HttpClient, private apiService: ApiService) { }

  /**
   * Método de inicialización del componente.
   * Se ejecuta cuando el componente se inicializa.
   * 
   * @returns {void}
   */
  ngOnInit(): void {
  }

  /**
 * Realiza una búsqueda en la API utilizando el término de búsqueda y los criterios definidos.
 * 
 * Este método construye el cuerpo de la solicitud para la búsqueda en la API,
 * utilizando el término de búsqueda actual (`this.searchTerm`) y la columna especificada (`this.columnName`).
 * Luego, envía una solicitud POST a la API y procesa los resultados.
 *
 * @returns {void}
 * @memberof AgAutocompleteComponent
 */
  busqueda(): void { //Nombre inconsistente
  const apiUrl = `${environment.URL_BACKEND}/api/search/${this.tableParam}`;
  const body = {
    page: 1,
    criteria_group: {
      group_list: [
        {
          logic_operator: "AND", // Ajusta esto según tus necesidades
          criteria_list: [
            {
              column_filter: this.columnName,
              compare_criteria: "PARECIDO A", // Ajusta esto según tus necesidades
              compare_value: this.searchTerm
            }
          ],
          group_list: []
        }
      ]
    }
  };

    this.apiService.post(apiUrl, body).subscribe(
      (data: any) => {
        this.results = data.results; // Ajusta según la estructura de respuesta de tu API
        this.filtrarResults();
        console.log(this.results);
      }
    );
  }

  /**
 * Filtra los resultados de la búsqueda en función del término de búsqueda.
 * 
 * Este método convierte el término de búsqueda (`this.searchTerm`) a minúsculas,
 * y luego filtra los resultados (`this.results`) para incluir solo aquellos
 * cuyo valor en la columna especificada (`this.columnName`) contiene el término de búsqueda.
 *
 * @returns {void}
 * @memberof AgAutocompleteComponent
 */
  filtrarResults(): void { //Nombre inconsistente
    const lowerCaseSearchTerm = this.searchTerm.toLowerCase();

    this.filteredResults = this.results.filter(item => {
      const value = item[this.columnName];
      return value && value.toLowerCase().includes(lowerCaseSearchTerm);
    });
  }

  /**
 * Maneja los cambios en el término de búsqueda y actualiza los resultados filtrados.
 * 
 * Este método verifica si el término de búsqueda (`this.searchTerm`) no está vacío ni contiene solo espacios en blanco.
 * Si se cumple esta condición, llama al método `busqueda()` para realizar una búsqueda.
 * Si el término de búsqueda está vacío, se restablece `this.filteredResults` a un arreglo vacío.
 *
 * @returns {void}
 * @memberof AgAutocompleteComponent
 */
  onInputChange(): void {
    if (this.searchTerm && this.searchTerm.trim() !== '') {
      this.busqueda();
    } else {
      this.filteredResults = [];
    }
  }

  /**
 * Maneja la selección de un resultado de búsqueda.
 * Establece el término de búsqueda (`this.searchTerm`) al valor del resultado seleccionado
 * en la columna especificada por `this.columnName`.
 * Luego, limpia la lista de resultados filtrados (`this.filteredResults`).
 *
 * @param {any} result - El resultado seleccionado que contiene el valor para el término de búsqueda.
 * @returns {void}
 */
selectResult(result: any): void {
  this.searchTerm = result[this.columnName];
  this.filteredResults = []; // Limpiar resultados después de seleccionar
  }
}

<div class="m-4">
  <input class="modern-input w-100" [(ngModel)]="searchTerm" (input)="onInputChange()" placeholder="Buscar..." />
  <ul class="autocomplete-results">
    <li class="autocomplete-item" (click)="selectResult(result)" *ngFor="let result of this.filteredResults">{{result[columnName]}}</li>
  </ul>
</div>
