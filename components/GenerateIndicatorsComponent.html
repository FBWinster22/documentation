import { Component } from '@angular/core';
import { AgFormComponent } from '../../../widgets/ag-form/ag-form.component';
import { environment } from '../../../../../environments/environment.development';
import { AgChartComponent } from '../../../widgets/ag-chart/ag-chart.component';
import { CommonModule } from '@angular/common';

/**
 * Componente Angular para la generación de gráficos basados en indicadores.
 * 
 * Este componente utiliza los widgets `AgFormComponent` y `AgChartComponent` para proporcionar
 * funcionalidades de formularios y gráficos dentro de la aplicación.
 * 
 * @component
 * @memberof GenerateIndicatorsComponent
 */
@Component({
  selector: 'app-ag-indicators-charts',
  standalone: true,
  imports: [CommonModule, AgFormComponent, AgChartComponent],
  templateUrl: './generate-indicators.component.html',
  styleUrl: './generate-indicators.component.css'
})
export class GenerateIndicatorsComponent {

  /**
   * Datos preparados para ser utilizados en gráficos.
   * 
   * Almacena los datos que se utilizarán para la visualización de gráficos, organizados por etiquetas
   * y datos asociados en un formato adecuado.
   * 
   * @type {{ label: string; data: { [key: string]: number } }[]}
   * @memberof GenerateIndicatorsComponent
   */
  dataParaPreGrafico: { label: string; data: { [key: string]: number } }[] = []; //Nombre inconsistente

  /**
   * Entorno de la aplicación.
   * 
   * Proporciona acceso a las variables y configuraciones del entorno de la aplicación.
   * 
   * @type {any}
   * @memberof GenerateIndicatorsComponent
   */
  environment = environment

  /**
   * Datos recibidos desde el formulario.
   * 
   * Almacena los datos recibidos a través de la función `handleFormSubmitted`, que se espera que
   * contengan los indicadores necesarios para el procesamiento y visualización.
   * 
   * @type {any}
   * @memberof GenerateIndicatorsComponent
   */
  datosRecibidos:any; // Nombre inconsistente

  /**
   * Datos para el gráfico.
   * 
   * Almacena los datos en un formato clave-valor adecuado para la visualización en el gráfico.
   * 
   * @type {{ [key: string]: number }}
   * @memberof GenerateIndicatorsComponent
   */
  dataParaGrafico:{ [key: string]: number } = {};

  constructor() {
    // Inicializa dataParaPreGrafico aquí si es necesario
  }

  /**
   * Maneja la presentación de datos del formulario.
   * 
   * Procesa los datos enviados desde el formulario y los prepara para su visualización en gráficos.
   * 
   * @param {any} data - Datos recibidos desde el formulario. Debe incluir una propiedad `content`
   *                       con los indicadores a procesar.
   * @memberof GenerateIndicatorsComponent
   */
  handleFormSubmitted(data: any) {
    this.datosRecibidos = data.content['indicadores']
    this.processData(this.datosRecibidos)
    // Aquí puedes manejar los datos recibidos
  }

  /**
   * Procesa los datos de indicadores para convertirlos en un formato adecuado para los gráficos.
   * 
   * La función limpia y convierte los datos JSON recibidos en un formato clave-valor adecuado para la visualización.
   * 
   * @param {string[]} indicadores - Lista de indicadores en formato de cadena, que se espera contengan datos JSON.
   * @memberof GenerateIndicatorsComponent
   */
  processData(indicadores: string[]) {
    this.dataParaPreGrafico = indicadores.map(indicador => {
      // Separar la etiqueta del valor JSON
      const [label, jsonData] = indicador.split(' = ');

      // Intentar limpiar y parsear el valor JSON
      let cleanedData = jsonData
        .replace(/'/g, '"') // Reemplazar comillas simples con comillas dobles
        .replace(/np\.float64\(/g, '') // Eliminar np.float64(
        .replace(/\)/g, ''); // Eliminar los paréntesis de cierre

      try {
        const data = JSON.parse(cleanedData);

        let dataParaGrafico: { [key: string]: number } = {};

        if (data.records) {
          data.records.forEach((record: any) => {
            dataParaGrafico[record.description] = record.porc;
          });
        } else if (Array.isArray(data)) {
          data.forEach((record: any) => {
            dataParaGrafico[record.description] = record.percentage;
          });
        } else {
          // Cuando no hay records y data no es un array, es un solo valor
          dataParaGrafico[label] = parseFloat(jsonData);
        }

        return { label, data: dataParaGrafico };
      } catch (e) {
        // Si no es JSON, simplemente asignar el valor como un par clave-valor en un objeto
        return { label, data: { [label]: parseFloat(jsonData) } };
      }
    });

    console.log(this.dataParaPreGrafico)
  }
}
