import { Input, Output, EventEmitter, Directive, ViewChild} from '@angular/core';
import { AgTableComponent } from './ag-table/ag-table.component';
import { AgSelectComponent } from './ag-select/ag-select.component';


@Directive()
export abstract class BaseDynamicInputComponent {

  /**
   * @property {AgTableComponent | undefined} input_search_table - Referencia al componente `AgTableComponent` 
   * para manejar la tabla de búsqueda. Utilizado con `@ViewChild`.
   */
  @ViewChild('input_search_table') input_search_table?: AgTableComponent
    
  
  /**
   * @property {AgSelectComponent | undefined} selectRef - Referencia al componente `AgSelectComponent` 
   * para manejar la selección de ítems. Utilizado con `@ViewChild`.
   */
  @ViewChild('selectRef') selectRef?: AgSelectComponent

  /**
   * @property {any} item - Objeto que representa el ítem actual con el que se está trabajando.
   */
  @Input() item: any;

  /**
   * @property {any} form_object - Objeto de formulario que contiene los datos del formulario asociado.
   */
  @Input() form_object: any;


  /**
   * @property {number | undefined} i - Índice o identificador del ítem actual en el contexto del componente.
   */
  @Input() i: number | undefined;


  /**
   * @property {boolean | undefined} isSearchInputModalOpen - Indica si el modal de búsqueda está abierto (`true`) 
   * o cerrado (`false`).
   */
  @Input() isSearchInputModalOpen: boolean | undefined;


  /**
   * @property {boolean | undefined} mostrarTablaResultados - Controla la visualización de la tabla de resultados. 
   * 
   */
  @Input() mostrarTablaResultados: boolean | undefined;


  /**
   * @property {boolean | undefined} required - Indica si el ítem es obligatorio (`true`) o no (`false`).
   */
  @Input() required?:boolean 



  /**
   * @property {object | undefined} request_body - Cuerpo de la solicitud para enviar datos o parámetros 
   * al componente de selección.
   */
  @Input() request_body?: object;



  /**
    * @property {boolean | undefined} isValid - Indica si los datos del formulario son válidos (`true`) 
    * o no (`false`).
    */
  @Input() isValid?: boolean;



  /**
   * Vista previa de la imagen.
   * 
   * Esta propiedad opcional permite establecer una vista previa de la imagen que se mostrará en el componente.
   * Puede ser una URL (`string`), un `ArrayBuffer`, `null` o `undefined`. Esto es útil en casos donde se desea 
   * mostrar una imagen cargada previamente o seleccionada por el usuario antes de ser subida o procesada.
   * 
   * @type {string | ArrayBuffer | null | undefined}
   * @memberof AgDynamicInputComponent
  */
  @Input() imagePreview?: string | ArrayBuffer | null | undefined
    
  /**
    * @property {EventEmitter<any>} openSearchInputModal - Emisor de eventos para abrir el modal de búsqueda.
  */
  @Output() openSearchInputModal = new EventEmitter<any>();
  
  /**
    * @property {EventEmitter<number>} closeSearchInputModal - Emisor de eventos para cerrar el modal de búsqueda.
  */
  @Output() closeSearchInputModal = new EventEmitter<number>();
  
  /**
   * @property {EventEmitter<any>} update_api_search_interface - Emisor de eventos para actualizar la interfaz 
   * de búsqueda API con nuevos datos.
  */
  @Output() update_api_search_interface = new EventEmitter<any>();
  
  /**
   * @property {EventEmitter<any>} asignarValue - Emisor de eventos para asignar un valor a un ítem seleccionado. 
   * (Nombre de variable inconsistente)
  */
  @Output() asignarValue = new EventEmitter<any>(); // Nombre de variable inconsistente
  
  /**
   * @property {EventEmitter<any>} setSelectValue - Emisor de eventos para establecer el valor de selección en el componente.
  */
  @Output() setSelectValue = new EventEmitter<any>();
  
  /**
   * @property {EventEmitter<any>} setSelectCallback - Emisor de eventos para manejar el callback de selección.
  */
  @Output() setSelectCallback = new EventEmitter<any>();
  
  /**
   * @property {EventEmitter<any>} onFileChange - Emisor de eventos para manejar los cambios en los archivos cargados.
  */
  @Output() onFileChange = new EventEmitter<any>();
  
  /**
   * @property {EventEmitter<any>} removeImg - Emisor de eventos para eliminar una imagen.
  */
  @Output() removeImg = new EventEmitter<any>();
  
  /**
   * Grupo de criterios de búsqueda para la API.
   * 
   * Esta propiedad opcional define un conjunto de criterios de búsqueda que se utilizarán al interactuar con la API.
   * Puede incluir diferentes operadores lógicos y condiciones para filtrar los resultados.
   * El tipo de dato es flexible (`any`) para permitir diversas estructuras de criterios de búsqueda.
   * 
   * @type {any}
   * @memberof AgDynamicInputComponent
  */
  api_search_criteria_group?: any

  /**
    * @property {Console} console - Referencia a la consola del navegador para depuración.
  */
  console = console
    

  /**
   * @description Este método se ejecuta al inicializar el componente. Su propósito principal es
   * imprimir en la consola el objeto de formulario (`form_object`) para fines de depuración.
   * @returns {void}
   * @memberof AgDynamicInputComponent
  */
  ngOnInit(): void {
      console.log(this.form_object)
  }

  /**
   * @memberof AgDynamicInputComponent
   * @description Este método verifica si el tipo de dato del ítem (`item.type`) es `tinyint` y si es obligatorio (`item.required` igual a `1`). 
   * Si ambas condiciones se cumplen, asigna el valor `1` al campo correspondiente en el objeto de formulario (`form_object`) y retorna `true`. 
   * De lo contrario, retorna `false`.
   * @returns {boolean} Retorna `true` si el ítem es de tipo `tinyint` y es obligatorio, de lo contrario retorna `false`.
  */
  porDefectoTrue() {//nombre de variable inconsistente
    if (this.item.type === 'tinyint' && this.item.required === 1) {
        this.form_object[this.item.name] = 1;
        return true;
    }
    return false;
  }


  /**
   * @memberof AgDynamicInputComponent
   * @description Verifica si el tipo de dato proporcionado (`type`) es uno de los tipos numéricos predefinidos 
   * en la lista: `int`, `float`, `double`, o `decimal`. Utiliza el método `includes` para comprobar la presencia 
   * del tipo en esta lista.
   * @param {string} type - El tipo de dato a verificar.
   * @returns {boolean} Retorna `true` si el tipo es uno de los tipos numéricos predefinidos, de lo contrario `false`.
  */
  isNumberType(type: string): boolean {
    return ['int', 'float', 'double', 'decimal'].includes(type);
  }

  /**
 * @memberof AgDynamicInputComponent
 * @description Maneja la selección de un ítem. Emite un evento a través del `EventEmitter` `asignarValue` 
 * con un objeto que contiene el nombre del ítem seleccionado (`name`), el evento asociado (`event`), 
 * y las opciones analizadas (`parsedOptions`). También cierra el modal de búsqueda estableciendo 
 * `isSearchInputModalOpen` en `false`.
 * @param {any} event - El evento asociado con la selección del ítem.
 * @param {string} name - El nombre del ítem seleccionado.
 * @param {any} parsedOptions - Las opciones analizadas relacionadas con el ítem seleccionado.
 * @returns {void}
 */
  handleSelectedItem(event: any, name: string, parsedOptions: any) {
    this.asignarValue.emit({ name, event, parsedOptions });
    this.isSearchInputModalOpen = false
  }

  /**
 * @memberof AgDynamicInputComponent
 * @description Actualiza la tabla de búsqueda con los nuevos criterios de búsqueda y emite un evento 
 * a través del `EventEmitter` `update_api_search_interface` con el evento proporcionado (`event`) 
 * y el índice actual (`i`). 
 * 
 * - Asigna el criterio de búsqueda (`event`) a la propiedad `api_search_criteria_group` del componente 
 * `input_search_table` (si está definido).
 * - Llama al método `callSearch` en el componente `input_search_table` (si está definido) para realizar la búsqueda.
 * - Emite el evento `update_api_search_interface` con los datos del evento y el índice actual.
 * 
 * @param {any} event - El nuevo criterio de búsqueda a aplicar a la tabla.
 * @returns {void}
 */
  updateSearchTable(event: any){
    this.input_search_table!.api_search_criteria_group = event; 
    this.input_search_table?.callSearch()
    this.update_api_search_interface.emit({ event: event, index: this.i })
  }

  /**
 * @memberof AgDynamicInputComponent
 * @description Asigna el objeto `request_body` al componente `selectRef` (si está definido) 
 * y luego llama al método `populate` en dicho componente para cargar los datos necesarios.
 * 
 * - Establece la propiedad `request_body` del componente `selectRef` con el valor de `request_body`.
 * - Llama al método `populate` del componente `selectRef` para inicializar o actualizar los datos.
 * 
 * @returns {void}
 */
  populate(){
    this.selectRef!.request_body = this.request_body
    this.selectRef!.populate()
  }
}
